49static int binary_search(mrbc_kv_handle *kvh, mrbc_sym sym_id)
50{
51 int left = 0;
52 int right = kvh->n_stored - 1;
53 if( right < 0 ) return -1;
54
55 while( left < right ) {
56 int mid = (left + right) / 2;
57 if( kvh->data[mid].sym_id < sym_id ) {
58 left = mid + 1;
59 } else {
60 right = mid;
61 }
62 }
63
64 return left;
65}

199int mrbc_kv_set(mrbc_kv_handle *kvh, mrbc_sym sym_id, mrbc_value *set_val)
200{
201 int idx = binary_search(kvh, sym_id);
202 if( idx < 0 ) {
203 idx = 0;
204 goto INSERT_VALUE;
205 }
206
207 // replace value ?
208 if( kvh->data[idx].sym_id == sym_id ) {
209 mrbc_decref( &kvh->data[idx].value );
210 kvh->data[idx].value = *set_val;
211 return 0;
212 }
213
214 if( kvh->data[idx].sym_id < sym_id ) {
215 idx++;
216 }
217
218 INSERT_VALUE:
219 // need alloc?
220 if( kvh->data_size == 0 ) {
221 kvh->data = mrbc_alloc(kvh->vm, sizeof(mrbc_kv) * MRBC_KV_SIZE_INIT);
222 if( kvh->data == NULL ) return E_NOMEMORY_ERROR; // ENOMEM
223 kvh->data_size = MRBC_KV_SIZE_INIT;
224
225#if defined(MRBC_DEBUG)
226 memcpy( kvh->data->type, "KV", 2 );
227#endif
228
229 // need resize?
230 } else if( kvh->n_stored >= kvh->data_size ) {
231 if( mrbc_kv_resize(kvh, kvh->data_size + MRBC_KV_SIZE_INCREMENT) != 0 ) {
232 return E_NOMEMORY_ERROR; // ENOMEM
233 }
234 }
235
236 // need move data?
237 if( idx < kvh->n_stored ) {
238 int size = sizeof(mrbc_kv) * (kvh->n_stored - idx);
239 memmove( &kvh->data[idx+1], &kvh->data[idx], size );
240 }
241
242 kvh->data[idx].sym_id = sym_id;
243 kvh->data[idx].value = *set_val;
244 kvh->n_stored++;
245
246 return 0;
247}
