50mrbc_value mrbc_exception_new(struct VM *vm, struct RClass *exc_cls, const void *message, int len )
51{
52 // allocate memory.
53 mrbc_exception *ex = mrbc_alloc( vm, sizeof(mrbc_exception) );
54 if( !ex ) { // ENOMEM
55 return mrbc_nil_value();
56 }
57
58 MRBC_INIT_OBJECT_HEADER( ex, "EX" );
59 ex->cls = exc_cls;
60
61 // in case of no message.
62 if( !message ) {
63 ex->message = 0;
64 ex->message_size = 0;
65 goto RETURN;
66 }
67
68 // in case of message is ""
69 if( *(const char *)message == 0 ) {
70 ex->message = (const uint8_t *)"";
71 ex->message_size = 0;
72 goto RETURN;
73 }
74
75 // in case of message in ROM.
76 if( len == 0 ) {
77 ex->message = message;
78 ex->message_size = 0;
79 goto RETURN;
80 }
81
82 // else, copy the message.
83 uint8_t *buf = mrbc_alloc( vm, len+1 );
84 if( buf ) {
85 memcpy( buf, message, len );
86 buf[len] = 0;
87 ex->message_size = len;
88 } else {
89 ex->message_size = 0;
90 }
91 ex->message = buf;
92
93 RETURN:
94 return (mrbc_value){.tt = MRBC_TT_EXCEPTION, .exception = ex};
95}

146void mrbc_raise( struct VM *vm, struct RClass *exc_cls, const char *msg )
147{
148 if( vm ) {
149 vm->exception = mrbc_exception_new( vm, exc_cls ? exc_cls : MRBC_CLASS(RuntimeError), msg, 0 );
150 vm->flag_preemption = 2;
151
152 } else {
153 mrbc_printf("Exception: %s (%s)\n", msg ? msg : mrbc_symid_to_str(exc_cls->sym_id), mrbc_symid_to_str(exc_cls->sym_id));
154 }

165void mrbc_raisef( struct VM *vm, struct RClass *exc_cls, const char *fstr, ... )
166{
167 static const int MESSAGE_INI_LEN = 32;
168 va_list ap;
169 va_start( ap, fstr );
170
171 char *buf = 0;
172 if( vm ) buf = mrbc_alloc( vm, MESSAGE_INI_LEN );
173
174 if( buf ) {
175 mrbc_vasprintf( &buf, MESSAGE_INI_LEN, fstr, ap );
176 vm->exception = mrbc_exception_new_alloc( vm,
177 exc_cls ? exc_cls : MRBC_CLASS(RuntimeError),
178 buf, strlen(buf) );
179 vm->flag_preemption = 2;
180
181 } else {
182 // VM == NULL or ENOMEM
183 mrbc_print("Exception: ");
184 mrbc_vprintf( fstr, ap );
185 mrbc_printf(" (%s)\n", exc_cls ? mrbc_symid_to_str(exc_cls->sym_id) : "RuntimeError");
186 }
187
188 va_end( ap );
189}

231static void c_exception_new(struct VM *vm, mrbc_value v[], int argc)
232{
233 assert( mrbc_type(v[0]) == MRBC_TT_CLASS );
234
235 mrbc_value value;
236 if( argc == 1 && mrbc_type(v[1]) == MRBC_TT_STRING ) {
237 value = mrbc_exception_new(vm, v[0].cls, mrbc_string_cstr(&v[1]), mrbc_string_size(&v[1]));
238 } else {
239 value = mrbc_exception_new(vm, v[0].cls, NULL, 0);
240 }
241
242 SET_RETURN(value);
243}

249static void c_exception_message(struct VM *vm, mrbc_value v[], int argc)
250{
251 mrbc_value value;
252
253 if( v[0].exception->message ) {
254 value = mrbc_string_new( vm, v[0].exception->message, v[0].exception->message_size );
255 } else {
256 value = mrbc_string_new_cstr(vm, mrbc_symid_to_str(v->exception->cls->sym_id));
257 }
258
259 mrbc_decref( &v[0] );
260 v[0] = value;
261}
