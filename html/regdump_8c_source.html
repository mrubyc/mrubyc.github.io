123static void REGDUMP( const mrbc_vm *vm, const mrbc_value *regs, int n, int argc )
124{
125 if( !FLAG_REGDUMP ) return;
126
127 int r_ofs = regs - vm->regs;
128 int nest_level = 0;
129
130 for( const mrbc_callinfo *callinfo = vm->callinfo_tail; callinfo != NULL;
131 callinfo = callinfo->prev ) {
132 nest_level++;
133 }
134 mrbc_printf("===== REGDUMP nest:%d exc.tt:%d ==================\n",
135 nest_level, vm->exception.tt );
136
137 int i;
138 for( i = 0; i < n; i++ ) {
139 mrbc_printf("%c R[%d:%d] tt=%d : ", (i == argc)? '>' : ' ',
140 r_ofs+i, i, mrbc_type(regs[i]) );
141 mrbc_p( regs+i );
142 }
143
144}

147static void REGDUMP_STACK( const mrbc_vm *vm, int n )
148{
149#define MAX_NEST_LEVEL 20
150
151 if( !FLAG_REGDUMP ) return;
152
153 struct {
154 const mrbc_value *regs;
155 int regs_offset;
156 mrbc_callinfo *callinfo;
157 } calltree[MAX_NEST_LEVEL] = {0};
158
159 // 現在のネスト数を計算
160 int nest_level = 0;
161 for( mrbc_callinfo *callinfo = vm->callinfo_tail; callinfo != NULL;
162 callinfo = callinfo->prev ) {
163 nest_level++;
164 }
165 if( nest_level >= MAX_NEST_LEVEL ) {
166 mrbc_printf("Warning: nest level over.\n");
167 return;
168 }
169
170 // calltree構造体配列へ、各ネストの情報を格納
171 int i = nest_level;
172 for( mrbc_callinfo *callinfo = vm->callinfo_tail; callinfo != NULL;
173 callinfo = callinfo->prev, i-- ) {
174 calltree[i].regs = callinfo->cur_regs;
175 //calltree[i].regs_offset = callinfo->cur_regs - vm->regs + callinfo->reg_offset;
176 calltree[i].regs_offset = callinfo->reg_offset;
177 calltree[i].callinfo = callinfo;
178 }
179 //calltree[0].regs = vm->cur_regs;
180 calltree[0].regs = vm->regs;
181
182 // display CALLINFO
183 mrbc_printf("===== CALLINFO => ");
184 for( int i = 1; i <= nest_level; i++ ) {
185 mrbc_printf("[%p/%d] ", calltree[i].callinfo, calltree[i].regs_offset);
186 }
187 mrbc_print("\n");
188
189 // display REGISTER
190 int reg_max = calltree[nest_level].regs_offset + n;
191 for( int reg = 0; reg < reg_max; reg++ ) {
192 mrbc_printf(" $%04x", (uintptr_t)&vm->regs[reg] & 0xffff );
193 int nest = 0;
194 while( nest <= nest_level ) {
195 mrbc_printf(" R[%2d] ", reg - calltree[nest].regs_offset );
196
197 // 次のネストがない時
198 if( nest >= nest_level ) break;
199
200 // ネストはあるがまだ表示しない時
201 if( reg < calltree[nest+1].regs_offset ) break;
202
203 mrbc_print("---> ");
204 nest++;
205 }
206 mrbc_printf("(tt=%d) ", mrbc_type(vm->regs[reg]) );
207 mrbc_p( &vm->regs[reg] );
208 }
209
210 // display ope code
211 const char *s1 = "";
212 const char *s2 = "";
213 int op = *vm->inst;
214 if( 0x2d <= op && op <= 0x30 ) { // OP_SEND?
215 int b = vm->inst[2]; // (NOTE) not support OP_EXT
216 s1 = ":";
217 s2 = mrbc_irep_symbol_cstr(vm->cur_irep, b);
218 }
219
220 mrbc_printf("===== [ %03d %02x %s %s%s] nest:%d exc.tt:%d ================\n",
221 (int)(vm->inst - vm->cur_irep->inst), op, OP_TBL[op],
222 s1, s2, nest_level, vm->exception.tt );
223}
