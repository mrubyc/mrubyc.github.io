186 int mrbc_printf_main( mrbc_printf *pf )
187 {
188  int ch = -1;
189  pf->fmt = (struct RPrintfFormat){0};
190 
191  while( pf->p < pf->buf_end && (ch = *pf->fstr) != '\0' ) {
192  pf->fstr++;
193  if( ch == '%' ) {
194  if( *pf->fstr == '%' ) { // is "%%"
195  pf->fstr++;
196  } else {
197  goto PARSE_FLAG;
198  }
199  }
200  *pf->p++ = ch;
201  }
202  return -(ch != '\0');
203 
204 
205  PARSE_FLAG:
206  // parse format - '%' [flag] [width] [.precision] type
207  // e.g. "%05d"
208  while( (ch = *pf->fstr) ) {
209  switch( ch ) {
210  case '+': pf->fmt.flag_plus = 1; break;
211  case ' ': pf->fmt.flag_space = 1; break;
212  case '-': pf->fmt.flag_minus = 1; break;
213  case '0': pf->fmt.flag_zero = 1; break;
214  default : goto PARSE_WIDTH;
215  }
216  pf->fstr++;
217  }
218 
219  PARSE_WIDTH:
220  while( (void)(ch = *pf->fstr - '0'), (0 <= ch && ch <= 9)) { // isdigit()
221  pf->fmt.width = pf->fmt.width * 10 + ch;
222  pf->fstr++;
223  }
224  if( *pf->fstr == '.' ) {
225  pf->fstr++;
226  while( (void)(ch = *pf->fstr - '0'), (0 <= ch && ch <= 9)) {
227  pf->fmt.precision = pf->fmt.precision * 10 + ch;
228  pf->fstr++;
229  }
230  }
231  if( *pf->fstr ) pf->fmt.type = *pf->fstr++;
232 
233  return 1;
234 }

332 int mrbc_printf_int( mrbc_printf *pf, mrbc_int value, int base )
333 {
334  int sign = 0;
335  mrbc_int v = value;
336 
337  if( value < 0 ) {
338  sign = '-';
339  v = -value;
340  } else if( pf->fmt.flag_plus ) {
341  sign = '+';
342  } else if( pf->fmt.flag_space ) {
343  sign = ' ';
344  }
345 
346  if( pf->fmt.flag_minus || pf->fmt.width == 0 ) {
347  pf->fmt.flag_zero = 0; // disable zero padding if left align or width zero.
348  }
349 
350  // create string to allocated buffer
351  int alloc_size = 32+2; // int32 + terminate + 1
352  if( alloc_size < pf->fmt.precision + 1 ) alloc_size = pf->fmt.precision + 1;
353  assert( sizeof(mrbc_int) * 8 < alloc_size );
354 
355  char *buf = mrbc_raw_alloc( alloc_size );
356  if( buf == NULL ) return 0; // ENOMEM
357 
358  char *p = buf + alloc_size - 1;
359  *p = '\0';
360  do {
361  assert( p != buf );
362  int i = v % base;
363  *--p = i + ((i < 10)? '0' : 'a' - 10);
364  v /= base;
365  } while( v != 0 );
366 
367  // precision parameter
368  int precision_remain = (int)pf->fmt.precision - ((buf + alloc_size - 1) - p);
369  while( --precision_remain >= 0 ) {
370  *--p = '0';
371  }
372  pf->fmt.precision = 0;
373 
374  // decide pad character and output sign character
375  int ret;
376  int pad;
377  if( pf->fmt.flag_zero ) {
378  pad = '0';
379  if( sign ) {
380  *pf->p++ = sign;
381  if( pf->p >= pf->buf_end ) {
382  ret = -1;
383  goto DONE;
384  }
385  pf->fmt.width--;
386  }
387  } else {
388  pad = ' ';
389  if( sign ) *--p = sign;
390  }
391  ret = mrbc_printf_str( pf, p, pad );
392 
393  DONE:
394  mrbc_raw_free( buf );
395  return ret;
396 }

410 int mrbc_printf_bit( mrbc_printf *pf, mrbc_int value, int bit )
411 {
412  if( pf->fmt.flag_plus || pf->fmt.flag_space ) {
413  return mrbc_printf_int( pf, value, 1 << bit );
414  }
415 
416  if( pf->fmt.flag_minus || pf->fmt.width == 0 ) {
417  pf->fmt.flag_zero = 0; // disable zero padding if left align or width zero.
418  }
419  pf->fmt.precision = 0;
420 
421  mrbc_int v = value;
422  int offset_a = (pf->fmt.type == 'X') ? 'A' - 10 : 'a' - 10;
423  int mask = (1 << bit) - 1; // 0x0f, 0x07, 0x01
424  int mchar = mask + ((mask < 10)? '0' : offset_a);
425 
426  // create string to local buffer
427  char buf[40]; // > int32(bit) + '..f\0'
428  assert( sizeof(buf) > (sizeof(mrbc_int) * 8 + 4) );
429  char *p = buf + sizeof(buf) - 1;
430  *p = '\0';
431  int n;
432  do {
433  assert( p >= buf );
434  n = v & mask;
435  *--p = n + ((n < 10)? '0' : offset_a);
436  v >>= bit;
437  } while( v != 0 && v != -1 );
438 
439  // add "..f" for negative value?
440  // (note) '0' flag such as "%08x" is incompatible with ruby.
441  if( value < 0 && !pf->fmt.flag_zero ) {
442  if( n != mask ) *--p = mchar;
443  *--p = '.';
444  *--p = '.';
445  assert( p >= buf );
446  }
447 
448  // decide pad character and output sign character
449  int pad;
450  if( pf->fmt.flag_zero ) {
451  pad = (value < 0) ? mchar : '0';
452  } else {
453  pad = ' ';
454  }
455 
456  return mrbc_printf_str( pf, p, pad );
457 }
