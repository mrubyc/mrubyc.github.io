95 static int send_by_name( struct VM *vm, const char *method_name, mrbc_value *regs, int a, int c, int is_sendb )
96 {
97  mrbc_value *recv = &regs[a];
98

100  int bidx = a + c + 1;
101  if( !is_sendb ){
102  mrbc_decref( &regs[bidx] );
103  regs[bidx].tt = MRBC_TT_NIL;
104  }
105 
106  mrbc_sym sym_id = str_to_symid(method_name);
107  mrbc_class *cls = find_class_by_object(recv);
108  mrbc_proc *m = find_method_by_class(&cls, cls, sym_id);
109 
110  if( m == 0 ) {
111  console_printf("Undefined local variable or method '%s' for %s\n",
112  method_name, symid_to_str( cls->sym_id ));
113  return 1;
114  }
115 
116  // m is C func
117  if( m->c_func ) {
118  m->func(vm, regs + a, c);
119  if( m->func == c_proc_call ) return 0;
120  if( vm->exc != NULL || vm->exc_pending != NULL ) return 0;
121 
122  int release_reg = a+1;
123  while( release_reg <= bidx ) {
124  mrbc_decref_empty(&regs[release_reg]);
125  release_reg++;
126  }
127  return 0;
128  }
129 
130  // m is Ruby method.
131  // callinfo
132  mrbc_callinfo *callinfo = mrbc_push_callinfo(vm, sym_id, a, c);
133  callinfo->own_class = cls;
134 
135  // target irep
136  vm->pc_irep = m->irep;
137  vm->inst = m->irep->code;
138

140  vm->current_regs += a;
141 
142  return 0;

1221 {

1227 #define FLAG_DICT_PARAM (a & 0x2)
1228  int argc = vm->callinfo_tail->n_args;
1229 
1230  if( a & 0xffc ) { // check m2 and k parameter.
1231  console_printf("ArgumentError: not support m2 or keyword argument.\n");
1232  return 1; // raise?
1233  }
1234

1236  console_printf("ArgumentError: wrong number of arguments.\n");
1237  return 1; // raise?
1238  }
1239 
1240  // save proc (or nil) object.
1241  mrbc_value proc = regs[argc + 1];
1242  regs[argc + 1].tt = MRBC_TT_EMPTY;
1243 
1244  // support yield [...] pattern

1246  regs[1].tt == MRBC_TT_ARRAY && argc != m1 ) {
1247  mrbc_value argary = regs[1];

1258  }
1259 
1260  // dictionary parameter if exists.
1261  mrbc_value dict;
1262  if( FLAG_DICT_PARAM ) {
1263  if( (argc - m1) > 0 && regs[argc].tt == MRBC_TT_HASH ) {
1264  dict = regs[argc];
1265  regs[argc--].tt = MRBC_TT_EMPTY;
1266  } else {
1267  dict = mrbc_hash_new( vm, 0 );
1268  }
1269  }
1270 
1271  // rest parameter if exists.
1272  mrbc_value rest;
1273  if( FLAG_REST_PARAM ) {
1274  int rest_size = argc - m1 - o;
1275  if( rest_size < 0 ) rest_size = 0;
1276  rest = mrbc_array_new(vm, rest_size);
1277  if( !rest.array ) return 0; // ENOMEM raise?
1278

1280  int i;
1281  for( i = 0; i < rest_size; i++ ) {
1282  mrbc_array_push( &rest, &regs[rest_reg] );
1283  regs[rest_reg++].tt = MRBC_TT_EMPTY;
1284  }
1285  }
1286 
1287  // reorder arguments.
1288  int i;
1289  if( argc < m1 ) {
1290  for( i = argc+1; i <= m1; i++ ) {
1291  regs[i].tt = MRBC_TT_NIL;
1292  }
1293  } else {
1294  i = m1 + 1;
1295  }
1296  i += o;
1297  if( FLAG_REST_PARAM ) {
1298  regs[i++] = rest;
1299  }
1300  if( FLAG_DICT_PARAM ) {
1301  regs[i++] = dict;
1302  }
1303  if( argc >= i ) i = argc + 1;
1304  regs[i] = proc;
1305  vm->callinfo_tail->n_args = i;
1306 
1307  // prepare for get default arguments.
1308  int jmp_ofs = argc - m1;
1309  if( jmp_ofs > 0 ) {
1310  if( jmp_ofs > o ) {
1311  if( !FLAG_REST_PARAM && regs[0].tt != MRBC_TT_PROC ) {
1312  console_printf("ArgumentError: wrong number of arguments.\n");
1313  return 1; // raise?
1314  }
1315  jmp_ofs = o;
1316  }
1317  vm->inst += jmp_ofs * 3; // 3 = bytecode size of OP_JMP
1318  }
1319 
1320  return 0;
1321 #undef FLAG_REST_PARAM
1322 #undef FLAG_DICT_PARAM
1323 }
