80mrbc_value mrbc_array_new(struct VM *vm, int size)
81{
82 mrbc_value value = {.tt = MRBC_TT_ARRAY};
83
84 /*
85 Allocate handle and data buffer.
86 */
87 mrbc_array *h = mrbc_alloc(vm, sizeof(mrbc_array));
88 if( !h ) return value; // ENOMEM
89
90 mrbc_value *data = mrbc_alloc(vm, sizeof(mrbc_value) * size);
91 if( !data ) { // ENOMEM
92 mrbc_raw_free( h );
93 return value;
94 }
95
96 MRBC_INIT_OBJECT_HEADER( h, "AR" );
97 h->data_size = size;
98 h->n_stored = 0;
99 h->data = data;
100
101 value.array = h;
102 return value;
103}

175int mrbc_array_set(mrbc_value *ary, int idx, mrbc_value *set_val)
176{
177 mrbc_array *h = ary->array;
178
179 if( idx < 0 ) {
180 idx = h->n_stored + idx;
181 if( idx < 0 ) return E_INDEX_ERROR;
182 }
183
184 // need resize?
185 if( idx >= h->data_size && mrbc_array_resize(ary, idx + 1) != 0 ) {
186 return E_NOMEMORY_ERROR; // ENOMEM
187 }
188
189 if( idx < h->n_stored ) {
190 // release existing data.
191 mrbc_decref( &h->data[idx] );
192 } else {
193 // clear empty cells.
194 int i;
195 for( i = h->n_stored; i < idx; i++ ) {
196 h->data[i] = mrbc_nil_value();
197 }
198 h->n_stored = idx + 1;
199 }
200
201 h->data[idx] = *set_val;
202
203 return 0;
204}

328int mrbc_array_insert(mrbc_value *ary, int idx, mrbc_value *set_val)
329{
330 mrbc_array *h = ary->array;
331
332 if( idx < 0 ) {
333 idx = h->n_stored + idx + 1;
334 if( idx < 0 ) return E_INDEX_ERROR;
335 }
336
337 // need resize?
338 int size = 0;
339 if( idx >= h->data_size ) {
340 size = idx + 1;
341 } else if( h->n_stored >= h->data_size ) {
342 size = h->data_size + 1;
343 }
344 if( size && mrbc_array_resize(ary, size) != 0 ) {
345 return E_NOMEMORY_ERROR; // ENOMEM
346 }
347
348 // move datas.
349 if( idx < h->n_stored ) {
350 memmove(h->data + idx + 1, h->data + idx,
351 sizeof(mrbc_value) * (h->n_stored - idx));
352 }
353
354 // set data
355 h->data[idx] = *set_val;
356 h->n_stored++;
357
358 // clear empty cells if need.
359 if( idx >= h->n_stored ) {
360 int i;
361 for( i = h->n_stored-1; i < idx; i++ ) {
362 h->data[i] = mrbc_nil_value();
363 }
364 h->n_stored = idx + 1;
365 }
366
367 return 0;
368}

456 mrbc_value *p_max_value = h->data;
457

459 for( i = 1; i < h->n_stored; i++ ) {

461 p_min_value = &h->data[i];

464 p_max_value = &h->data[i];
465 }
466 }
467

512mrbc_value mrbc_array_divide(struct VM *vm, mrbc_value *src, int pos)
513{
514 mrbc_array *ha_s = src->array;
515 if( pos < 0 ) pos = 0;
516 int new_size = ha_s->n_stored - pos;
517 if( new_size < 0 ) new_size = 0;
518 int remain_size = ha_s->n_stored - new_size;
519
520 mrbc_value ret = mrbc_array_new(vm, new_size);
521 if( ret.array == NULL ) return ret; // ENOMEM
522 mrbc_array *ha_r = ret.array;
523

525 ha_s->n_stored = remain_size;
526 mrbc_array_resize( src, remain_size );
527 ha_r->n_stored = new_size;
528
529 return ret;
530}

536static void c_array_new(struct VM *vm, mrbc_value v[], int argc)
537{
538 /*
539 in case of new()
540 */
541 if( argc == 0 ) {
542 mrbc_value ret = mrbc_array_new(vm, 0);
543 if( ret.array == NULL ) return; // ENOMEM
544
545 SET_RETURN(ret);
546 return;
547 }
548
549 /*
550 in case of new(num)
551 */
552 if( argc == 1 && mrbc_type(v[1]) == MRBC_TT_INTEGER && mrbc_integer(v[1]) >= 0 ) {
553 int num = mrbc_integer(v[1]);
554 mrbc_value ret = mrbc_array_new(vm, num);
555 if( ret.array == NULL ) return; // ENOMEM
556
557 if( num > 0 ) {
558 mrbc_array_set(&ret, num - 1, &mrbc_nil_value());
559 }
560 SET_RETURN(ret);
561 return;
562 }
563
564 /*
565 in case of new(num, value)
566 */
567 if( argc == 2 && mrbc_type(v[1]) == MRBC_TT_INTEGER && mrbc_integer(v[1]) >= 0 ) {
568 int num = mrbc_integer(v[1]);
569 mrbc_value ret = mrbc_array_new(vm, num);
570 if( ret.array == NULL ) return; // ENOMEM
571
572 int i;
573 for( i = 0; i < num; i++ ) {
574 mrbc_incref(&v[2]);
575 mrbc_array_set(&ret, i, &v[2]);
576 }
577 SET_RETURN(ret);
578 return;
579 }
580
581 /*
582 other case
583 */
584 mrbc_raise( vm, MRBC_CLASS(ArgumentError), 0 );
585}

591static void c_array_add(struct VM *vm, mrbc_value v[], int argc)
592{
593 if( mrbc_type(v[1]) != MRBC_TT_ARRAY ) {
594 mrbc_raise( vm, MRBC_CLASS(TypeError), 0 );
595 return;
596 }
597
598 mrbc_array *h1 = v[0].array;
599 mrbc_array *h2 = v[1].array;
600
601 mrbc_value value = mrbc_array_new(vm, h1->n_stored + h2->n_stored);
602 if( value.array == NULL ) return; // ENOMEM
603
604 memcpy( value.array->data, h1->data,
605 sizeof(mrbc_value) * h1->n_stored );
606 memcpy( value.array->data + h1->n_stored, h2->data,
607 sizeof(mrbc_value) * h2->n_stored );
608 value.array->n_stored = h1->n_stored + h2->n_stored;
609
610 mrbc_value *p1 = value.array->data;
611 const mrbc_value *p2 = p1 + value.array->n_stored;
612 while( p1 < p2 ) {
613 mrbc_incref(p1++);
614 }
615
616 mrbc_decref_empty(v+1);
617 SET_RETURN(value);
618}

624static void c_array_get(struct VM *vm, mrbc_value v[], int argc)
625{
626 /*
627 in case of Array[...] -> Array
628 */
629 if( mrbc_type(v[0]) == MRBC_TT_CLASS ) {
630 mrbc_value ret = mrbc_array_new(vm, argc);
631 if( ret.array == NULL ) return; // ENOMEM
632
633 memcpy( ret.array->data, &v[1], sizeof(mrbc_value) * argc );
634 int i;
635 for( i = 1; i <= argc; i++ ) {
636 mrbc_type(v[i]) = MRBC_TT_EMPTY;
637 }
638 ret.array->n_stored = argc;
639
640 SET_RETURN(ret);
641 return;
642 }
643
644 /*
645 in case of self[nth] -> object | nil
646 */
647 if( argc == 1 && mrbc_type(v[1]) == MRBC_TT_INTEGER ) {
648 mrbc_value ret = mrbc_array_get(v, mrbc_integer(v[1]));
649 mrbc_incref(&ret);
650 SET_RETURN(ret);
651 return;
652 }
653
654 /*
655 in case of self[start, length] -> Array | nil
656 */
657 if( argc == 2 && mrbc_type(v[1]) == MRBC_TT_INTEGER && mrbc_type(v[2]) == MRBC_TT_INTEGER ) {
658 int len = mrbc_array_size(&v[0]);
659 int idx = mrbc_integer(v[1]);
660 if( idx < 0 ) idx += len;
661 if( idx < 0 ) goto RETURN_NIL;
662
663 int size = (mrbc_integer(v[2]) < (len - idx)) ? mrbc_integer(v[2]) : (len - idx);
664 // min( mrbc_integer(v[2]), (len - idx) )
665 if( size < 0 ) goto RETURN_NIL;
666
667 mrbc_value ret = mrbc_array_new(vm, size);
668 if( ret.array == NULL ) return; // ENOMEM
669
670 int i;
671 for( i = 0; i < size; i++ ) {
672 mrbc_value val = mrbc_array_get(v, mrbc_integer(v[1]) + i);
673 mrbc_incref(&val);
674 mrbc_array_push(&ret, &val);
675 }
676
677 SET_RETURN(ret);
678 return;
679 }
680
681 /*
682 other case
683 */
684 mrbc_raise( vm, MRBC_CLASS(ArgumentError), 0 );
685 return;
686
687 RETURN_NIL:
688 SET_NIL_RETURN();
689}

695static void c_array_set(struct VM *vm, mrbc_value v[], int argc)
696{
697 /*
698 in case of self[nth] = val
699 */
700 if( argc == 2 && mrbc_type(v[1]) == MRBC_TT_INTEGER ) {
701 if( mrbc_array_set(v, mrbc_integer(v[1]), &v[2]) != 0 ) {
702 mrbc_raise( vm, MRBC_CLASS(IndexError), "too small for array");
703 }
704 mrbc_type(v[2]) = MRBC_TT_EMPTY;
705 return;
706 }
707
708 /*
709 in case of self[start, length] = val
710 */
711 if( argc == 3 && mrbc_type(v[1]) == MRBC_TT_INTEGER && mrbc_type(v[2]) == MRBC_TT_INTEGER ) {
712 int pos = mrbc_integer(v[1]);
713 int len = mrbc_integer(v[2]);
714
715 if( pos < 0 ) {
716 pos = 0;
717 } else if( pos > v[0].array->n_stored ) {
718 mrbc_array_set( &v[0], pos-1, &mrbc_nil_value() );
719 len = 0;
720 }
721 if( len < 0 ) len = 0;
722 if( pos+len > v[0].array->n_stored ) {
723 len = v[0].array->n_stored - pos;
724 }
725
726 // split 2 part
727 mrbc_value v1 = mrbc_array_divide(vm, &v[0], pos+len);
728 mrbc_array *ha0 = v[0].array;
729
730 // delete data from tail.
731 int i;
732 for( i = 0; i < len; i++ ) {
733 mrbc_decref( &ha0->data[--ha0->n_stored] );
734 }
735
736 // append data
737 if( v[3].tt == MRBC_TT_ARRAY ) {
738 mrbc_array_push_m(&v[0], &v[3]);
739 for( i = 0; i < v[3].array->n_stored; i++ ) {
740 mrbc_incref( &v[3].array->data[i] );
741 }
742 } else {
743 mrbc_array_push(&v[0], &v[3]);
744 v[3].tt = MRBC_TT_EMPTY;
745 }
746
747 mrbc_array_push_m(&v[0], &v1);
748 mrbc_array_delete_handle( &v1 );
749 return;
750 }
751
752 /*
753 other case
754 */
755 mrbc_raise( vm, MRBC_CLASS(ArgumentError), 0 );
756}
