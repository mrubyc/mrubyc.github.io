139 static mrbc_int shift(mrbc_int x, mrbc_int y)
140 {
141  // Don't support environments that include padding in int.
142  const int INT_BITS = sizeof(mrbc_int) * CHAR_BIT;
143 
144  if( y >= INT_BITS ) return 0;
145  if( y >= 0 ) return x << y;
146  if( y <= -INT_BITS ) return 0;
147  return x >> -y;
148 }
