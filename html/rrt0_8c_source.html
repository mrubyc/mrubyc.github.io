70static void q_insert_task(mrbc_tcb *p_tcb)
71{
72 mrbc_tcb **pp_q;
73
74 switch( p_tcb->state ) {
75 case TASKSTATE_DORMANT: pp_q = &q_dormant_; break;
76 case TASKSTATE_READY:
77 case TASKSTATE_RUNNING: pp_q = &q_ready_; break;
78 case TASKSTATE_WAITING: pp_q = &q_waiting_; break;
79 case TASKSTATE_SUSPENDED: pp_q = &q_suspended_; break;
80 default:
81 assert(!"Wrong task state.");
82 return;
83 }
84
85 // case insert on top.
86 if((*pp_q == NULL) ||
87 (p_tcb->priority_preemption < (*pp_q)->priority_preemption)) {
88 p_tcb->next = *pp_q;
89 *pp_q = p_tcb;
90 assert(p_tcb->next != p_tcb);
91 return;
92 }
93
94 // find insert point in sorted linked list.
95 mrbc_tcb *p = *pp_q;
96 while( 1 ) {
97 if((p->next == NULL) ||
98 (p_tcb->priority_preemption < p->next->priority_preemption)) {
99 p_tcb->next = p->next;
100 p->next = p_tcb;
101 assert(p->next != p);
102 return;
103 }
104
105 p = p->next;
106 }
107}

118static void q_delete_task(mrbc_tcb *p_tcb)
119{
120 mrbc_tcb **pp_q;
121
122 switch( p_tcb->state ) {
123 case TASKSTATE_DORMANT: pp_q = &q_dormant_; break;
124 case TASKSTATE_READY:
125 case TASKSTATE_RUNNING: pp_q = &q_ready_; break;
126 case TASKSTATE_WAITING: pp_q = &q_waiting_; break;
127 case TASKSTATE_SUSPENDED: pp_q = &q_suspended_; break;
128 default:
129 assert(!"Wrong task state.");
130 return;
131 }
132
133 if( *pp_q == NULL ) return;
134 if( *pp_q == p_tcb ) {
135 *pp_q = p_tcb->next;
136 p_tcb->next = NULL;
137 return;
138 }
139
140 mrbc_tcb *p = *pp_q;
141 while( p ) {
142 if( p->next == p_tcb ) {
143 p->next = p_tcb->next;
144 p_tcb->next = NULL;
145 return;
146 }
147
148 p = p->next;
149 }
150}

334void mrbc_tick(void)
335{
336 mrbc_tcb *tcb;
337 int flag_preemption = 0;
338
339 tick_++;
340
341 // 実行中タスクのタイムスライス値を減らす
342 tcb = q_ready_;
343 if((tcb != NULL) &&
344 (tcb->state == TASKSTATE_RUNNING) &&
345 (tcb->timeslice > 0)) {
346 tcb->timeslice--;
347 if( tcb->timeslice == 0 ) tcb->vm.flag_preemption = 1;
348 }
349
350 // 待ちタスクキューから、ウェイクアップすべきタスクを探す
351 tcb = q_waiting_;
352 while( tcb != NULL ) {
353 mrbc_tcb *t = tcb;
354 tcb = tcb->next;
355
356 if( t->reason == TASKREASON_SLEEP && t->wakeup_tick == tick_ ) {
357 q_delete_task(t);
358 t->state = TASKSTATE_READY;
359 t->timeslice = MRBC_TIMESLICE_TICK_COUNT;
360 q_insert_task(t);
361 flag_preemption = 1;
362 }
363 }
364
365 if( flag_preemption ) {
366 tcb = q_ready_;
367 while( tcb != NULL ) {
368 if( tcb->state == TASKSTATE_RUNNING ) tcb->vm.flag_preemption = 1;
369 tcb = tcb->next;
370 }
371 }
372}

451mrbc_tcb* mrbc_create_task(const uint8_t *vm_code, mrbc_tcb *tcb)
452{
453 // allocate Task Control Block
454 if( tcb == NULL ) {
455 tcb = mrbc_raw_alloc( sizeof(mrbc_tcb) );
456 if( tcb == NULL ) return NULL; // ENOMEM
457
458 mrbc_init_tcb( tcb );
459 }
460 tcb->timeslice = MRBC_TIMESLICE_TICK_COUNT;
461 tcb->priority_preemption = tcb->priority;
462
463 // assign VM ID
464 if( mrbc_vm_open( &tcb->vm ) == NULL ) {
465 mrbc_printf("Error: Can't assign VM-ID.\n");
466 return NULL;
467 }
468
469 if( mrbc_load_mrb(&tcb->vm, vm_code) != 0 ) {
470 mrbc_print_vm_exception( &tcb->vm );
471 mrbc_vm_close( &tcb->vm );
472 return NULL;
473 }
474 if( tcb->state != TASKSTATE_DORMANT ) {
475 mrbc_vm_begin( &tcb->vm );
476 }
477
478 hal_disable_irq();
479 q_insert_task(tcb);
480 hal_enable_irq();
481
482 return tcb;
483}

493{

516//================================================================
520int mrbc_run(void)
521{
522 int ret = 1;
523
524#if MRBC_SCHEDULER_EXIT
525 if( q_ready_ == NULL && q_waiting_ == NULL && q_suspended_ == NULL ) return 0;
526#endif
527
528 while( 1 ) {
529 mrbc_tcb *tcb = q_ready_;
530 if( tcb == NULL ) { // no task to run.
531 hal_idle_cpu();
532 continue;
533 }
534
535 tcb->state = TASKSTATE_RUNNING; // to execute.
536 int res = 0;
537
538#ifndef MRBC_NO_TIMER
539 tcb->vm.flag_preemption = 0;
540 res = mrbc_vm_run(&tcb->vm);
541
542#else
543 while( tcb->timeslice > 0 ) {
544 tcb->vm.flag_preemption = 1;
545 res = mrbc_vm_run(&tcb->vm);
546 tcb->timeslice--;
547 if( res != 0 ) break;
548 if( tcb->state != TASKSTATE_RUNNING ) break;
549 }
550 mrbc_tick();
551#endif /* ifndef MRBC_NO_TIMER */
552
553 // did the task done?
554 if( res != 0 ) {
555 hal_disable_irq();
556 q_delete_task(tcb);
557 tcb->state = TASKSTATE_DORMANT;
558 q_insert_task(tcb);
559 hal_enable_irq();
560 if( tcb->vm.flag_permanence == 0 ) mrbc_vm_end(&tcb->vm);
561 if( res != 1 ) ret = res;
562
563#if MRBC_SCHEDULER_EXIT
564 if( q_ready_ == NULL && q_waiting_ == NULL && q_suspended_ == NULL ) break;
565#endif
566 continue;
567 }
568
569 // switch task.
570 hal_disable_irq();
571 if( tcb->state == TASKSTATE_RUNNING ) {
572 tcb->state = TASKSTATE_READY;
573
574 if( tcb->timeslice == 0 ) {
575 q_delete_task(tcb);
576 tcb->timeslice = MRBC_TIMESLICE_TICK_COUNT;
577 q_insert_task(tcb); // insert task on queue last.
578 }
579 }
580 hal_enable_irq();
581 }
582
583 return ret;
584}

591void mrbc_sleep_ms(mrbc_tcb *tcb, uint32_t ms)
592{
593 hal_disable_irq();

595 tcb->timeslice = 0;
596 tcb->state = TASKSTATE_WAITING;
597 tcb->reason = TASKREASON_SLEEP;

600 q_insert_task(tcb);
601 hal_enable_irq();
602
603 tcb->vm.flag_preemption = 1;
604}

690int mrbc_mutex_lock( mrbc_mutex *mutex, mrbc_tcb *tcb )
691{
692 MRBC_MUTEX_TRACE("mutex lock / MUTEX: %p TCB: %p", mutex, tcb );
693
694 int ret = 0;
695 hal_disable_irq();
696
697 // Try lock mutex;
698 if( mutex->lock == 0 ) { // a future does use TAS?
699 mutex->lock = 1;
700 mutex->tcb = tcb;
701 MRBC_MUTEX_TRACE(" lock OK\n" );
702 goto DONE;
703 }
704 MRBC_MUTEX_TRACE(" lock FAIL\n" );
705
706 // Can't lock mutex
707 // check recursive lock.
708 if( mutex->tcb == tcb ) {
709 ret = 1;
710 goto DONE;
711 }
712
713 // To WAITING state.
714 q_delete_task(tcb);
715 tcb->state = TASKSTATE_WAITING;
716 tcb->reason = TASKREASON_MUTEX;
717 tcb->mutex = mutex;
718 q_insert_task(tcb);
719 tcb->vm.flag_preemption = 1;
720
721 DONE:
722 hal_enable_irq();
723
724 return ret;
725}

728//================================================================
732int mrbc_mutex_unlock( mrbc_mutex *mutex, mrbc_tcb *tcb )
733{
734 MRBC_MUTEX_TRACE("mutex unlock / MUTEX: %p TCB: %p\n", mutex, tcb );
735
736 // check some parameters.
737 if( !mutex->lock ) return 1;
738 if( mutex->tcb != tcb ) return 2;
739
740 // wakeup ONE waiting task.
741 int flag_preemption = 0;
742 hal_disable_irq();
743 tcb = q_waiting_;
744 while( tcb != NULL ) {
745 if( tcb->reason == TASKREASON_MUTEX && tcb->mutex == mutex ) {
746 MRBC_MUTEX_TRACE("SW: TCB: %p\n", tcb );
747 mutex->tcb = tcb;
748 q_delete_task(tcb);
749 tcb->state = TASKSTATE_READY;
750 q_insert_task(tcb);
751 flag_preemption = 1;
752 break;
753 }
754 tcb = tcb->next;
755 }
756
757 if( flag_preemption ) {
758 tcb = q_ready_;
759 while( tcb != NULL ) {
760 if( tcb->state == TASKSTATE_RUNNING ) tcb->vm.flag_preemption = 1;
761 tcb = tcb->next;
762 }
763 }
764 else {
765 // unlock mutex
766 MRBC_MUTEX_TRACE("mutex unlock all.\n" );
767 mutex->lock = 0;
768 }
769
770 hal_enable_irq();
771
772 return 0;
773}
