202 static inline int nlz16(uint16_t x)
203 {
204  if( x == 0 ) return 16;
205 
206  int n = 1;
207  if((x >> 8) == 0 ) { n += 8; x <<= 8; }
208  if((x >> 12) == 0 ) { n += 4; x <<= 4; }
209  if((x >> 14) == 0 ) { n += 2; x <<= 2; }
210  return n - (x >> 15);
211 }

237 static unsigned int calc_index(unsigned int alloc_size)
238 {
239  // check overflow
240  if( (alloc_size >> (MRBC_ALLOC_FLI_BIT_WIDTH
241  + MRBC_ALLOC_SLI_BIT_WIDTH
242  + MRBC_ALLOC_IGNORE_LSBS)) != 0) {
243  return SIZE_FREE_BLOCKS;
244  }
245 
246  // calculate First Level Index.
247  int fli = 16 -
248  nlz16( alloc_size >> (MRBC_ALLOC_SLI_BIT_WIDTH + MRBC_ALLOC_IGNORE_LSBS) );
249 
250  // calculate Second Level Index.
251  int shift = (fli == 0) ? MRBC_ALLOC_IGNORE_LSBS :
252  (MRBC_ALLOC_IGNORE_LSBS - 1 + fli);
253 
254  int sli = (alloc_size >> shift) & ((1 << MRBC_ALLOC_SLI_BIT_WIDTH) - 1);
255  int index = (fli << MRBC_ALLOC_SLI_BIT_WIDTH) + sli;
256 
257  assert(fli >= 0);
258  assert(fli <= MRBC_ALLOC_FLI_BIT_WIDTH);
259  assert(sli >= 0);
260  assert(sli <= (1 << MRBC_ALLOC_SLI_BIT_WIDTH) - 1);
261 
262  return index;
263 }

444 void * mrbc_raw_alloc(unsigned int size)
445 {
446  unsigned int alloc_size = size + sizeof(USED_BLOCK);
447 
448  // align 4 byte
449  alloc_size += (-alloc_size & 3);
450 
451  // check minimum alloc size.
452  if( alloc_size < MRBC_MIN_MEMORY_BLOCK_SIZE ) alloc_size = MRBC_MIN_MEMORY_BLOCK_SIZE;
453 
454  // find free memory block.
455  unsigned int index = calc_index(alloc_size);
456  int fli = FLI(index);
457  int sli = SLI(index);
458 
459  FREE_BLOCK *target = free_blocks[index];
460  if( target != NULL ) goto FOUND_TARGET_BLOCK;
461 
462  // check in SLI bitmap table.
463  uint16_t masked = free_sli_bitmap[fli] & ((MSB_BIT1_SLI >> sli) - 1);
464  if( masked != 0 ) {
465  sli = NLZ_SLI( masked );
466  goto FOUND_FLI_SLI;
467  }
468 
469  // check in FLI bitmap table.
470  masked = free_fli_bitmap & ((MSB_BIT1_FLI >> fli) - 1);
471  if( masked != 0 ) {
472  fli = NLZ_FLI( masked );
473  sli = NLZ_SLI( free_sli_bitmap[fli] );
474  goto FOUND_FLI_SLI;
475  }
476 
477  // Change strategy to First-fit.
478  target = mrbc_raw_alloc_ff_sub( alloc_size, index );
479  if( target ) goto SPLIT_BLOCK;
480 
481  // else out of memory
482  static const char msg[] = "Fatal error: Out of memory.\n";
483  hal_write(1, msg, sizeof(msg)-1);
484  return NULL; // ENOMEM
485 
486 
487  FOUND_FLI_SLI:
488  assert(fli >= 0);
489  assert(fli <= MRBC_ALLOC_FLI_BIT_WIDTH);
490  assert(sli >= 0);
491  assert(sli <= (1 << MRBC_ALLOC_SLI_BIT_WIDTH) - 1);
492 
493  index = (fli << MRBC_ALLOC_SLI_BIT_WIDTH) + sli;
494  target = free_blocks[index];
495  assert( target != NULL );
496 
497  FOUND_TARGET_BLOCK:
498  assert(BLOCK_SIZE(target) >= alloc_size);
499 
500  // remove free_blocks index
501  free_blocks[index] = target->next_free;
502  if( target->next_free == NULL ) {
503  free_sli_bitmap[fli] &= ~(MSB_BIT1_SLI >> sli);
504  if( free_sli_bitmap[fli] == 0 ) free_fli_bitmap &= ~(MSB_BIT1_FLI >> fli);
505  }
506  else {
507  target->next_free->prev_free = NULL;
508  }
509 
510  SPLIT_BLOCK: {
511  FREE_BLOCK *release = split_block(target, alloc_size);
512  if( release != NULL ) {
513  SET_PREV_USED(release);
514  add_free_block(release);
515  } else {
516  FREE_BLOCK *next = PHYS_NEXT(target);
517  SET_PREV_USED(next);
518  }
519  }
520 
521  SET_USED_BLOCK(target);
522 #if defined(MRBC_ALLOC_VMID)
523  target->vm_id = 0;
524 #endif
525 
526 #ifdef MRBC_DEBUG
527  memset( (uint8_t *)target + sizeof(USED_BLOCK), 0xaa,
528  BLOCK_SIZE(target) - sizeof(USED_BLOCK) );
529 #endif
530 
531  return (uint8_t *)target + sizeof(USED_BLOCK);
532 }

542 void * mrbc_raw_alloc_no_free(unsigned int size)
543 {
544  unsigned int alloc_size = size + (-size & 3); // align 4 byte
545 
546  // find the tail block
547  FREE_BLOCK *tail = (FREE_BLOCK *)memory_pool;
548  FREE_BLOCK *prev;
549  do {
550  prev = tail;
551  tail = PHYS_NEXT(tail);
552  } while( PHYS_NEXT(tail) < (void *)(memory_pool + memory_pool_size) );
553 
554  // can resize it block?
555  if( IS_USED_BLOCK(prev) ) goto FALLBACK;
556  if( (BLOCK_SIZE(prev) - sizeof(USED_BLOCK)) < size ) goto FALLBACK;
557 
558  remove_free_block( prev );
559  unsigned int free_size = BLOCK_SIZE(prev) - alloc_size;
560 
561  if( free_size <= MRBC_MIN_MEMORY_BLOCK_SIZE ) {
562  // no split, use all
563  prev->size += BLOCK_SIZE(tail);
564  SET_USED_BLOCK( prev );
565  tail = prev;
566  }
567  else {
568  // split block
569  unsigned int tail_size = tail->size + alloc_size; // w/ flags.
570  tail = (FREE_BLOCK*)((uint8_t *)tail - alloc_size);
571  tail->size = tail_size;
572  prev->size -= alloc_size; // w/ flags.
573  add_free_block( prev );
574  }
575 
576  return (uint8_t *)tail + sizeof(USED_BLOCK);
577 
578  FALLBACK:

588 void mrbc_raw_free(void *ptr)
589 {
590  // get target block
591  FREE_BLOCK *target = (FREE_BLOCK *)((uint8_t *)ptr - sizeof(USED_BLOCK));
592 
593  // check next block, merge?
594  FREE_BLOCK *next = PHYS_NEXT(target);
595 
596  if( IS_FREE_BLOCK(next) ) {
597  remove_free_block(next);
598  merge_block(target, next);
599  } else {
600  SET_PREV_FREE(next);
601  }
602 
603  if( IS_PREV_FREE(target) ) {
604  FREE_BLOCK *prev = *((FREE_BLOCK **)((uint8_t*)target - sizeof(FREE_BLOCK *)));
605 
606  assert( IS_FREE_BLOCK(prev) );
607  remove_free_block(prev);
608  merge_block(prev, target);
609  target = prev;
610  }
611 
612  // target, add to index
613  add_free_block(target);
614 }

625 void * mrbc_raw_realloc(void *ptr, unsigned int size)
626 {
627  USED_BLOCK *target = (USED_BLOCK *)((uint8_t *)ptr - sizeof(USED_BLOCK));
628  unsigned int alloc_size = size + sizeof(USED_BLOCK);
629  FREE_BLOCK *next;
630 
631  // align 4 byte
632  alloc_size += (-alloc_size & 3);
633 
634  // check minimum alloc size.
635  if( alloc_size < MRBC_MIN_MEMORY_BLOCK_SIZE ) alloc_size = MRBC_MIN_MEMORY_BLOCK_SIZE;
636 
637  // expand? part1.
638  // next phys block is free and enough size?
639  if( alloc_size > BLOCK_SIZE(target) ) {
640  next = PHYS_NEXT(target);
641  if( IS_USED_BLOCK(next) ) goto ALLOC_AND_COPY;
642  if( (BLOCK_SIZE(target) + BLOCK_SIZE(next)) < alloc_size ) goto ALLOC_AND_COPY;
643 
644  remove_free_block(next);
645  merge_block((FREE_BLOCK *)target, next);
646  }
647  next = PHYS_NEXT(target);
648 
649  // try shrink.
650  FREE_BLOCK *release = split_block((FREE_BLOCK *)target, alloc_size);
651  if( release != NULL ) {
652  SET_PREV_USED(release);
653  } else {
654  SET_PREV_USED(next);
655  return ptr;
656  }
657 
658  // check next block, merge?
659  if( IS_FREE_BLOCK(next) ) {
660  remove_free_block(next);
661  merge_block(release, next);
662  } else {
663  SET_PREV_FREE(next);
664  }
665  add_free_block(release);
666  return ptr;
667 
668 
669  // expand part2.
670  // new alloc and copy
671  ALLOC_AND_COPY: {
672  uint8_t *new_ptr = mrbc_raw_alloc(size);
673  if( new_ptr == NULL ) return NULL; // ENOMEM
674 
675  memcpy(new_ptr, ptr, BLOCK_SIZE(target) - sizeof(USED_BLOCK));
676  SET_VM_ID(new_ptr, target->vm_id);
677 
678  mrbc_raw_free(ptr);
679 
680  return new_ptr;
681  }
682 }
