50static int is_space( int ch )
51{
52 static const char ws[] = " \t\r\n\f\v"; // '\0' on tail
53
54 int i;
55 for( i = 0; i < sizeof(ws); i++ ) {

71mrbc_value mrbc_string_new(struct VM *vm, const void *src, int len)
72{
73 mrbc_value value = {.tt = MRBC_TT_STRING};
74
75 /*
76 Allocate handle and string buffer.
77 */
78 mrbc_string *h = mrbc_alloc(vm, sizeof(mrbc_string));
79 if( !h ) return value; // ENOMEM
80
81 uint8_t *str = mrbc_alloc(vm, len+1);
82 if( !str ) { // ENOMEM
83 mrbc_raw_free( h );
84 return value;
85 }
86
87 MRBC_INIT_OBJECT_HEADER( h, "ST" );
88 h->size = len;
89 h->data = str;
90
91 /*
92 Copy a source string.
93 */
94 if( src == NULL ) {
95 str[0] = '\0';
96 } else {
97 memcpy( str, src, len );
98 str[len] = '\0';
99 }
100
101 value.string = h;
102 return value;
103}

286int mrbc_string_index(const mrbc_value *src, const mrbc_value *pattern, int offset)
287{
288 char *p1 = mrbc_string_cstr(src) + offset;
289 char *p2 = mrbc_string_cstr(pattern);
290 int try_cnt = mrbc_string_size(src) - mrbc_string_size(pattern) - offset;
291
292 while( try_cnt >= 0 ) {
293 if( memcmp( p1, p2, mrbc_string_size(pattern) ) == 0 ) {
294 return p1 - mrbc_string_cstr(src); // matched.
295 }
296 try_cnt--;
297 p1++;
298 }
299
300 return -1;
301}

311int mrbc_string_strip(mrbc_value *src, int mode)
312{
313 char *p1 = mrbc_string_cstr(src);
314 char *p2 = p1 + mrbc_string_size(src) - 1;
315
316 // left-side
317 if( mode & 0x01 ) {
318 while( p1 <= p2 ) {
319 if( *p1 == '\0' ) break;
320 if( !is_space(*p1) ) break;
321 p1++;
322 }
323 }
324 // right-side
325 if( mode & 0x02 ) {
326 while( p1 <= p2 ) {
327 if( !is_space(*p2) ) break;
328 p2--;
329 }
330 }
331
332 int new_size = p2 - p1 + 1;
333 if( mrbc_string_size(src) == new_size ) return 0;
334
335 char *buf = mrbc_string_cstr(src);
336 if( p1 != buf ) memmove( buf, p1, new_size );
337 buf[new_size] = '\0';
338 mrbc_raw_realloc(buf, new_size+1); // shrink suitable size.
339 src->string->size = new_size;
340
341 return 1;
342}

547{
548 int nth;
549 int len;
550 const mrbc_value *val;
551
552 /*
553 in case of self[nth] = val
554 */
555 if( argc == 2 && mrbc_type(v[1]) == MRBC_TT_INTEGER &&
556 mrbc_type(v[2]) == MRBC_TT_STRING ) {
557 nth = v[1].i;
558 len = 1;
559 val = &v[2];
560 }
561 /*
562 in case of self[nth, len] = val
563 */
564 else if( argc == 3 && mrbc_type(v[1]) == MRBC_TT_INTEGER &&
565 mrbc_type(v[2]) == MRBC_TT_INTEGER &&
566 mrbc_type(v[3]) == MRBC_TT_STRING ) {
567 nth = v[1].i;
568 len = v[2].i;
569 val = &v[3];
570 }
571 /*
572 other cases
573 */
574 else {
575 mrbc_raise( vm, MRBC_CLASS(TypeError), "Not supprted." );
576 return;
577 }
578
579 int len1 = v->string->size;
580 int len2 = val->string->size;
581 if( nth < 0 ) nth = len1 + nth; // adjust to positive number.
582 if( len > len1 - nth ) len = len1 - nth;
583 if( nth < 0 || nth > len1 || len < 0) {
584 mrbc_raisef( vm, MRBC_CLASS(IndexError), "index %d out of string", nth );
585 return;
586 }
587
588 uint8_t *str = mrbc_realloc(vm, mrbc_string_cstr(v), len1 + len2 - len + 1);
589 if( !str ) return;
590
591 memmove( str + nth + len2, str + nth + len, len1 - nth - len + 1 );
592 memcpy( str + nth, mrbc_string_cstr(val), len2 );
593 v->string->size = len1 + len2 - len;
594
595 v->string->data = str;
596}

792static void c_string_split(struct VM *vm, mrbc_value v[], int argc)
793{
794 mrbc_value ret = mrbc_array_new(vm, 0);
795 if( mrbc_string_size(&v[0]) == 0 ) goto DONE;
796
797 // check limit parameter.
798 int limit = 0;
799 if( argc >= 2 ) {
800 if( mrbc_type(v[2]) != MRBC_TT_INTEGER ) {
801 mrbc_raise( vm, MRBC_CLASS(ArgumentError), 0 );
802 return;
803 }
804 limit = v[2].i;
805 if( limit == 1 ) {
806 mrbc_array_push( &ret, &v[0] );
807 mrbc_incref( &v[0] );
808 goto DONE;
809 }
810 }
811
812 // check separator parameter.
813 mrbc_value sep = (argc == 0) ? mrbc_string_new_cstr(vm, " ") : v[1];
814 switch( mrbc_type(sep) ) {
815 case MRBC_TT_NIL:
816 sep = mrbc_string_new_cstr(vm, " ");
817 break;
818
819 case MRBC_TT_STRING:
820 break;
821
822 default:
823 mrbc_raise( vm, MRBC_CLASS(TypeError), 0 );
824 return;
825 }
826
827 int flag_strip = (mrbc_string_cstr(&sep)[0] == ' ') &&
828 (mrbc_string_size(&sep) == 1);
829 int offset = 0;
830 int sep_len = mrbc_string_size(&sep);
831 if( sep_len == 0 ) sep_len++;
832
833 while( 1 ) {
834 int pos, len = 0;
835
836 if( flag_strip ) {
837 for( ; offset < mrbc_string_size(&v[0]); offset++ ) {
838 if( !is_space( mrbc_string_cstr(&v[0])[offset] )) break;
839 }
840 if( offset > mrbc_string_size(&v[0])) break;
841 }
842
843 // check limit
844 if( limit > 0 && mrbc_array_size(&ret)+1 >= limit ) {
845 pos = -1;
846 goto SPLIT_ITEM;
847 }
848
849 // split by space character.
850 if( flag_strip ) {
851 pos = offset;
852 for( ; pos < mrbc_string_size(&v[0]); pos++ ) {
853 if( is_space( mrbc_string_cstr(&v[0])[pos] )) break;
854 }
855 len = pos - offset;
856 goto SPLIT_ITEM;
857 }
858
859 // split by each character.
860 if( mrbc_string_size(&sep) == 0 ) {
861 pos = (offset < mrbc_string_size(&v[0])-1) ? offset : -1;
862 len = 1;
863 goto SPLIT_ITEM;
864 }
865
866 // split by specified character.
867 pos = mrbc_string_index( &v[0], &sep, offset );
868 len = pos - offset;
869
870
871 SPLIT_ITEM:
872 if( pos < 0 ) len = mrbc_string_size(&v[0]) - offset;
873
874 mrbc_value v1 = mrbc_string_new(vm, mrbc_string_cstr(&v[0]) + offset, len);
875 mrbc_array_push( &ret, &v1 );
876

1014static struct tr_pattern * tr_parse_pattern( struct VM *vm, const mrbc_value *v_pattern, int flag_reverse_enable )
1015{
1016 const char *pattern = mrbc_string_cstr( v_pattern );
1017 int pattern_length = mrbc_string_size( v_pattern );
1018 int flag_reverse = 0;
1019 struct tr_pattern *ret = NULL;
1020
1021 int i = 0;
1022 if( flag_reverse_enable && pattern_length >= 2 && pattern[i] == '^' ) {
1023 flag_reverse = 1;
1024 i++;
1025 }
1026
1027 struct tr_pattern *pat1;
1028 while( i < pattern_length ) {
1029 // is range pattern ?
1030 if( (i+2) < pattern_length && pattern[i+1] == '-' ) {
1031 pat1 = mrbc_alloc( vm, sizeof(struct tr_pattern) + 2 );
1032 if( pat1 != NULL ) {
1033 pat1->type = 2;
1034 pat1->flag_reverse = flag_reverse;
1035 pat1->n = pattern[i+2] - pattern[i] + 1;
1036 pat1->next = NULL;
1037 pat1->ch[0] = pattern[i];
1038 pat1->ch[1] = pattern[i+2];
1039 }
1040 i += 3;
1041
1042 } else {
1043 // in order pattern.
1044 int start_pos = i++;
1045 while( i < pattern_length ) {
1046 if( (i+2) < pattern_length && pattern[i+1] == '-' ) break;
1047 i++;
1048 }
1049
1050 int len = i - start_pos;
1051 pat1 = mrbc_alloc( vm, sizeof(struct tr_pattern) + len );
1052 if( pat1 != NULL ) {
1053 pat1->type = 1;
1054 pat1->flag_reverse = flag_reverse;
1055 pat1->n = len;
1056 pat1->next = NULL;
1057 memcpy( pat1->ch, &pattern[start_pos], len );
1058 }
1059 }
1060
1061 // connect linked list.
1062 if( ret == NULL ) {
1063 ret = pat1;
1064 } else {
1065 struct tr_pattern *p = ret;
1066 while( p->next != NULL ) { p = p->next; }
1067 p->next = pat1;
1068 }
1069 }
1070
1071 return ret;
1072}

1074static int tr_find_character( const struct tr_pattern *pat, int ch )
1075{
1076 int ret = -1;
1077 int n_sum = 0;
1078 int flag_reverse = pat ? pat->flag_reverse : 0;
1079
1080 while( pat != NULL ) {
1081 if( pat->type == 1 ) { // in-order
1082 int i;
1083 for( i = 0; i < pat->n; i++ ) {
1084 if( pat->ch[i] == ch ) ret = n_sum + i;
1085 }
1086 } else { // pat->type == 2 range
1087 if( pat->ch[0] <= ch && ch <= pat->ch[1] ) ret = n_sum + ch - pat->ch[0];
1088 }
1089 n_sum += pat->n;
1090 pat = pat->next;
1091 }
1092
1093 if( flag_reverse ) {
1094 return (ret < 0) ? INT_MAX : -1;
1095 }
1096 return ret;
1097}

1099static int tr_get_character( const struct tr_pattern *pat, int n_th )
1100{
1101 int n_sum = 0;
1102 while( pat != NULL ) {
1103 if( n_th < (n_sum + pat->n) ) {
1104 int i = (n_th - n_sum);
1105 return (pat->type == 1) ? pat->ch[i] :pat->ch[0] + i;
1106 }
1107 if( pat->next == NULL ) {
1108 return (pat->type == 1) ? pat->ch[pat->n - 1] : pat->ch[1];
1109 }
1110 n_sum += pat->n;
1111 pat = pat->next;
1112 }
1113
1114 return -1;
1115}

1117static int tr_main( struct VM *vm, mrbc_value v[], int argc )
1118{
1119 if( !(argc == 2 && mrbc_type(v[1]) == MRBC_TT_STRING &&
1120 mrbc_type(v[2]) == MRBC_TT_STRING)) {
1121 mrbc_raise( vm, MRBC_CLASS(ArgumentError), 0 );
1122 return -1;
1123 }
1124
1125 struct tr_pattern *pat = tr_parse_pattern( vm, &v[1], 1 );
1126 if( pat == NULL ) return 0;
1127
1128 struct tr_pattern *rep = tr_parse_pattern( vm, &v[2], 0 );
1129
1130 int flag_changed = 0;
1131 char *s = mrbc_string_cstr( &v[0] );
1132 int len = mrbc_string_size( &v[0] );
1133 int i;
1134 for( i = 0; i < len; i++ ) {
1135 int n = tr_find_character( pat, s[i] );

1137
1138 flag_changed = 1;
1139 if( rep == NULL ) {
1140 memmove( s + i, s + i + 1, len - i );
1141 len--;
1142 i--;
1143 } else {
1144 s[i] = tr_get_character( rep, n );
1145 }
1146 }
1147
1148 tr_free_pattern( pat );
1149 tr_free_pattern( rep );
1150
1151 v[0].string->size = len;
1152 v[0].string->data[len] = 0;
1153
1154 return flag_changed;
1155}
