704 static void c_string_split(struct VM *vm, mrbc_value v[], int argc)
705 {
706  mrbc_value ret = mrbc_array_new(vm, 0);
707  if( mrbc_string_size(&v[0]) == 0 ) goto DONE;
708 
709  // check limit parameter.
710  int limit = 0;
711  if( argc >= 2 ) {
712  if( v[2].tt != MRBC_TT_FIXNUM ) {
713  console_print( "TypeError\n" ); // raise?
714  return;
715  }
716  limit = v[2].i;
717  if( limit == 1 ) {
718  mrbc_array_push( &ret, &v[0] );
719  mrbc_incref( &v[0] );
720  goto DONE;
721  }
722  }
723 
724  // check separator parameter.
725  mrb_value sep = (argc == 0) ? mrbc_string_new_cstr(vm, " ") : v[1];
726  switch( sep.tt ) {
727  case MRBC_TT_NIL:
728  sep = mrbc_string_new_cstr(vm, " ");
729  break;
730 
731  case MRBC_TT_STRING:
732  break;
733 
734  default:
735  console_print( "TypeError\n" ); // raise?
736  return;
737  }
738 
739  int flag_strip = (mrbc_string_cstr(&sep)[0] == ' ') &&
740  (mrbc_string_size(&sep) == 1);
741  int offset = 0;
742  int sep_len = mrbc_string_size(&sep);
743  if( sep_len == 0 ) sep_len++;
744 
745  while( 1 ) {
746  int pos, len = 0;
747 
748  if( flag_strip ) {
749  for( ; offset < mrbc_string_size(&v[0]); offset++ ) {
750  if( !is_space( mrbc_string_cstr(&v[0])[offset] )) break;
751  }
752  if( offset > mrbc_string_size(&v[0])) break;
753  }
754 
755  // check limit
756  if( limit > 0 && mrbc_array_size(&ret)+1 >= limit ) {
757  pos = -1;
758  goto SPLIT_ITEM;
759  }
760 
761  // split by space character.
762  if( flag_strip ) {
763  pos = offset;
764  for( ; pos < mrbc_string_size(&v[0]); pos++ ) {
765  if( is_space( mrbc_string_cstr(&v[0])[pos] )) break;
766  }
767  len = pos - offset;
768  goto SPLIT_ITEM;
769  }
770 
771  // split by each character.
772  if( mrbc_string_size(&sep) == 0 ) {
773  pos = (offset < mrbc_string_size(&v[0])-1) ? offset : -1;
774  len = 1;
775  goto SPLIT_ITEM;
776  }
777 
778  // split by specified character.
779  pos = mrbc_string_index( &v[0], &sep, offset );
780  len = pos - offset;
781 
782 
783  SPLIT_ITEM:
784  if( pos < 0 ) len = mrbc_string_size(&v[0]) - offset;
785 
786  mrb_value v1 = mrbc_string_new(vm, mrbc_string_cstr(&v[0]) + offset, len);
787  mrbc_array_push( &ret, &v1 );
788 
789  if( pos < 0 ) break;
790  offset = pos + sep_len;
791  }
792 
793  // remove trailing empty item
794  if( limit == 0 ) {
795  while( 1 ) {
796  int idx = mrbc_array_size(&ret) - 1;
797  if( idx < 0 ) break;
798 
799  mrb_value v1 = mrbc_array_get( &ret, idx );
800  if( mrbc_string_size(&v1) != 0 ) break;
801 
802  mrbc_array_remove(&ret, idx);
803  mrbc_string_delete( &v1 );
804  }
805  }
806 
807  if( argc == 0 || v[1].tt == MRBC_TT_NIL ) {
808  mrbc_string_delete(&sep);
809  }
810 
811  DONE:
812  SET_RETURN( ret );
813 }

926 static struct tr_pattern * tr_parse_pattern( struct VM *vm, const mrb_value *v_pattern, int flag_reverse_enable )
927 {
928  const char *pattern = mrbc_string_cstr( v_pattern );
929  int pattern_length = mrbc_string_size( v_pattern );
930  int flag_reverse = 0;
931  struct tr_pattern *ret = NULL;
932 
933  int i = 0;
934  if( flag_reverse_enable && pattern_length >= 2 && pattern[i] == '^' ) {
935  flag_reverse = 1;
936  i++;
937  }
938 
939  struct tr_pattern *pat1;
940  while( i < pattern_length ) {
941  // is range pattern ?
942  if( (i+2) < pattern_length && pattern[i+1] == '-' ) {
943  pat1 = mrbc_alloc( vm, sizeof(struct tr_pattern) + 2 );
944  if( pat1 != NULL ) {
945  pat1->type = 2;
946  pat1->flag_reverse = flag_reverse;
947  pat1->n = pattern[i+2] - pattern[i] + 1;
948  pat1->next = NULL;
949  pat1->ch[0] = pattern[i];
950  pat1->ch[1] = pattern[i+2];
951  }
952  i += 3;
953 
954  } else {
955  // in order pattern.
956  int start_pos = i++;
957  while( i < pattern_length ) {
958  if( (i+2) < pattern_length && pattern[i+1] == '-' ) break;
959  i++;
960  }
961 
962  int len = i - start_pos;
963  pat1 = mrbc_alloc( vm, sizeof(struct tr_pattern) + len );
964  if( pat1 != NULL ) {
965  pat1->type = 1;
966  pat1->flag_reverse = flag_reverse;
967  pat1->n = len;
968  pat1->next = NULL;
969  memcpy( pat1->ch, &pattern[start_pos], len );
970  }
971  }
972 
973  // connect linked list.
974  if( ret == NULL ) {
975  ret = pat1;
976  } else {
977  struct tr_pattern *p = ret;
978  while( p->next != NULL ) { p = p->next; }
979  p->next = pat1;
980  }
981  }
982 
983  return ret;
984 }

986 static int tr_find_character( const struct tr_pattern *pat, int ch )
987 {
988  int ret = -1;
989  int n_sum = 0;
990  int flag_reverse = pat ? pat->flag_reverse : 0;
991 
992  while( pat != NULL ) {
993  if( pat->type == 1 ) { // in-order
994  int i;
995  for( i = 0; i < pat->n; i++ ) {
996  if( pat->ch[i] == ch ) ret = n_sum + i;
997  }
998  } else { // pat->type == 2 range
999  if( pat->ch[0] <= ch && ch <= pat->ch[1] ) ret = n_sum + ch - pat->ch[0];
1000  }
1001  n_sum += pat->n;
1002  pat = pat->next;
1003  }
1004 
1005  if( flag_reverse ) {
1006  return (ret < 0) ? INT_MAX : -1;
1007  }
1008  return ret;
1009 }

1011 static int tr_get_character( const struct tr_pattern *pat, int n_th )
1012 {
1013  int n_sum = 0;
1014  while( pat != NULL ) {
1015  if( n_th < (n_sum + pat->n) ) {
1016  int i = (n_th - n_sum);
1017  return (pat->type == 1) ? pat->ch[i] :pat->ch[0] + i;
1018  }
1019  if( pat->next == NULL ) {
1020  return (pat->type == 1) ? pat->ch[pat->n - 1] : pat->ch[1];
1021  }
1022  n_sum += pat->n;
1023  pat = pat->next;
1024  }
1025 
1026  return -1;
1027 }
