77static inline uint16_t calc_hash(const char *str)
78{
79 uint16_t h = 0;
80
81 while( *str != '\0' ) {
82 h = h * 17 + *str++;
83 }

94static int search_builtin_symbol( const char *str )
95{
96 int left = 0;
97 int right = sizeof(builtin_symbols) / sizeof(builtin_symbols[0]);
98
99 while( left < right ) {
100 int mid = (left + right) / 2;
101 const unsigned char *p1 = (const unsigned char *)builtin_symbols[mid];
102 const unsigned char *p2 = (const unsigned char *)str;
103
104 while( 1 ) { // string compare, same order as cruby.
105 if( *p1 < *p2 ) {
106 left = mid + 1;
107 break;
108 }
109 if( *p1 > *p2 ) {
110 right = mid;
111 break;
112 }
113 if( *p1 == 0 ) {
114 return mid;
115 }
116
117 p1++;
118 p2++;
119 }
120 }
121
122 return -1;
123}

133static int search_index( uint16_t hash, const char *str )
134{
135#ifdef MRBC_SYMBOL_SEARCH_LINER
136 int i;
137 for( i = 0; i < sym_index_pos; i++ ) {
138 if( sym_index[i].hash == hash && strcmp(str, sym_index[i].cstr) == 0 ) {
139 return i;
140 }
141 }
142 return -1;
143#endif
144
145#ifdef MRBC_SYMBOL_SEARCH_BTREE
146 int i = 0;
147 do {
148 if( sym_index[i].hash == hash && strcmp(str, sym_index[i].cstr) == 0 ) {
149 return i;
150 }
151 if( hash < sym_index[i].hash ) {
152 i = sym_index[i].left;
153 } else {
154 i = sym_index[i].right;
155 }
156 } while( i != 0 );
157 return -1;
158#endif
159}

169static int add_index( uint16_t hash, const char *str )
170{
171 if( sym_index_pos >= MAX_SYMBOLS_COUNT ) return -1; // check overflow.
172
173 int idx = sym_index_pos++;
174
175 // append table.
176 sym_index[idx].hash = hash;
177 sym_index[idx].cstr = str;
178
179#ifdef MRBC_SYMBOL_SEARCH_BTREE
180 int i = 0;
181
182 while( 1 ) {
183 if( hash < sym_index[i].hash ) {
184 // left side
185 if( sym_index[i].left == 0 ) { // left is empty?
186 sym_index[i].left = idx;
187 break;
188 }
189 i = sym_index[i].left;
190 } else {
191 // right side
192 if( sym_index[i].right == 0 ) { // right is empty?
193 sym_index[i].right = idx;
194 break;
195 }
196 i = sym_index[i].right;
197 }
198 }
199#endif
200
201 return idx;
202}

223mrbc_sym mrbc_str_to_symid(const char *str)
224{
225 mrbc_sym sym_id = search_builtin_symbol(str);
226 if( sym_id >= 0 ) return sym_id;
227
228 uint16_t h = calc_hash(str);
229 sym_id = search_index(h, str);
230 if( sym_id < 0 ) sym_id = add_index( h, str );
231 if( sym_id < 0 ) return sym_id;
232
233 return sym_id + OFFSET_BUILTIN_SYMBOL;
234}

284mrbc_value mrbc_symbol_new(struct VM *vm, const char *str)
285{
286 mrbc_sym sym_id = mrbc_search_symid( str );
287 if( sym_id >= 0 ) goto DONE;
288
289 // create symbol object dynamically.
290 int size = strlen(str) + 1;
291 char *buf = mrbc_raw_alloc_no_free(size);
292 if( buf == NULL ) return mrbc_nil_value(); // ENOMEM raise?
293
294 memcpy(buf, str, size);
295 sym_id = add_index( calc_hash(buf), buf );
296 if( sym_id < 0 ) {
297 mrbc_raisef(vm, MRBC_CLASS(Exception),
298 "Overflow MAX_SYMBOLS_COUNT for '%s'", str );
299 return mrbc_nil_value();
300 }
301
302 sym_id += OFFSET_BUILTIN_SYMBOL;
303
304 DONE:
305 return mrbc_symbol_value( sym_id );
306}

314static void c_symbol_all_symbols(struct VM *vm, mrbc_value v[], int argc)
315{
