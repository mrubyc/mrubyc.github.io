173/*
174 (note)
175 Conversion functions from binary (byte array) to each data type.
176
177 Use the MRBC_BIG_ENDIAN, MRBC_LITTLE_ENDIAN and MRBC_REQUIRE_32BIT_ALIGNMENT
178 macros.
179
180 (each cases)
181 Little endian, no alignment.
182 MRBC_LITTLE_ENDIAN && !MRBC_REQUIRE_32BIT_ALIGNMENT
183 (e.g.) ARM Cortex-M4, Intel x86
184
185 Big endian, no alignment.
186 MRBC_BIG_ENDIAN && !MRBC_REQUIRE_32BIT_ALIGNMENT
187 (e.g.) IBM PPC405
188
189 Little endian, 32bit alignment required.
190 MRBC_LITTLE_ENDIAN && MRBC_REQUIRE_32BIT_ALIGNMENT
191 (e.g.) ARM Cortex-M0
192
193 Big endian, 32bit alignment required.
194 MRBC_BIG_ENDIAN) && MRBC_REQUIRE_32BIT_ALIGNMENT
195 (e.g.) OpenRISC
196*/

208static inline uint16_t bin_to_uint16( const void *s )
209{
210#if defined(MRBC_LITTLE_ENDIAN) && !defined(MRBC_REQUIRE_32BIT_ALIGNMENT)
211 // Little endian, no alignment.
212 uint16_t x = *((uint16_t *)s);
213 x = (x << 8) | (x >> 8);
214
215#elif defined(MRBC_BIG_ENDIAN) && !defined(MRBC_REQUIRE_32BIT_ALIGNMENT)
216 // Big endian, no alignment.
217 uint16_t x = *((uint16_t *)s);
218
219#elif defined(MRBC_REQUIRE_32BIT_ALIGNMENT)
220 // 32bit alignment required.
221 uint8_t *p = (uint8_t *)s;
222 uint16_t x = *p++;
223 x <<= 8;
224 x |= *p;
225
226#else
227 #error "Specify MRBC_BIG_ENDIAN or MRBC_LITTLE_ENDIAN"
228#endif
229
230 return x;
231}

240static inline uint32_t bin_to_uint32( const void *s )
241{
242#if defined(MRBC_LITTLE_ENDIAN) && !defined(MRBC_REQUIRE_32BIT_ALIGNMENT)
243 // Little endian, no alignment.
244 uint32_t x = *((uint32_t *)s);
245 x = (x << 24) | ((x & 0xff00) << 8) | ((x >> 8) & 0xff00) | (x >> 24);
246
247#elif defined(MRBC_BIG_ENDIAN) && !defined(MRBC_REQUIRE_32BIT_ALIGNMENT)
248 // Big endian, no alignment.
249 uint32_t x = *((uint32_t *)s);
250
251#elif defined(MRBC_REQUIRE_32BIT_ALIGNMENT)
252 // 32bit alignment required.
253 uint8_t *p = (uint8_t *)s;
254 uint32_t x = *p++;
255 x <<= 8;
256 x |= *p++;
257 x <<= 8;
258 x |= *p++;
259 x <<= 8;
260 x |= *p;
261
262#endif
263
264 return x;
265}

326static inline double bin_to_double64( const void *s )
327{
328#if defined(MRBC_LITTLE_ENDIAN) && !defined(MRBC_REQUIRE_64BIT_ALIGNMENT)
329 // Little endian, no alignment.
330 return *((double *)s);
331
332#elif defined(MRBC_BIG_ENDIAN) && !defined(MRBC_REQUIRE_64BIT_ALIGNMENT)
333 // Big endian, no alignment.
334 double x;
335 uint8_t *p1 = (uint8_t*)s;
336 uint8_t *p2 = (uint8_t*)&x + 7;
337 int i;
338 for( i = 7; i >= 0; i-- ) {
339 *p2-- = *p1++;
340 }
341 return x;
342
343#elif defined(MRBC_LITTLE_ENDIAN) && defined(MRBC_REQUIRE_64BIT_ALIGNMENT)
344 // Little endian, 64bit alignment required.

347 uint8_t *p2 = (uint8_t*)&x;

349 for( i = 7; i >= 0; i-- ) {
