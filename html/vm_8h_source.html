125  // Little endian, no alignment.
126  // e.g. ARM Coretex-M4, Intel x86
127 #if defined(MRBC_LITTLE_ENDIAN) && !defined(MRBC_REQUIRE_32BIT_ALIGNMENT)
128  uint32_t x = *((uint32_t *)s);
129  return (x << 24) | ((x & 0xff00) << 8) | ((x >> 8) & 0xff00) | (x >> 24);
130 
131  // Big endian, no alignment.
132  // e.g. IBM PPC405
133 #elif defined(MRBC_BIG_ENDIAN) && !defined(MRBC_REQUIRE_32BIT_ALIGNMENT)
134  uint32_t x = *((uint32_t *)s);
135  return x;
136 
137  // 32bit alignment required.
138  // Little endian
139  // e.g. ARM Coretex-M0
140  // Big endian
141  // e.g. OpenRISC
142 #elif defined(MRBC_REQUIRE_32BIT_ALIGNMENT)
143  uint8_t *p = (uint8_t *)s;
144  uint32_t x = *p++;
145  x <<= 8;
146  x |= *p++;
147  x <<= 8;
148  x |= *p++;
149  x <<= 8;
150  x |= *p;
