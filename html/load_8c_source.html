80static int load_header(struct VM *vm, const uint8_t *bin)
81{
82 if( memcmp(bin, RITE, sizeof(RITE)) != 0 ) {
83 mrbc_raise( vm, MRBC_CLASS(Exception), "Illegal bytecode.");
84 return -1;
85 }
86 bin += sizeof(RITE);
87
88 if( memcmp(bin, RITE_VERSION, sizeof(RITE_VERSION)) != 0 ) {
89 mrbc_raise( vm, MRBC_CLASS(Exception), "Bytecode version mismatch.");
90 return -1;
91 }
92
93 /* Ignore others. */
94
95 return 0;
96}

137static mrbc_irep * load_irep_1(struct VM *vm, const uint8_t *bin, int *len, int flag_top)
138{
139 mrbc_irep irep;
140 const uint8_t *p = bin + 4; // 4 = skip record size.
141 int i;
142
143#if defined(MRBC_DEBUG)
144 irep.type[0] = 'R'; // set "RP"
145 irep.type[1] = 'P';
146#endif
147
148 irep.nlocals = bin_to_uint16(p); p += 2;
149 irep.nregs = bin_to_uint16(p); p += 2;
150 irep.rlen = bin_to_uint16(p); p += 2;
151 irep.clen = bin_to_uint16(p); p += 2;
152 irep.ilen = bin_to_uint32(p); p += 4;
153 irep.inst = p;
154
155 // POOL block
156 p += irep.ilen + SIZE_RITE_CATCH_HANDLER * irep.clen;
157 irep.pool = p;
158 irep.plen = bin_to_uint16(p); p += 2;
159
160 // skip pool
161 for( i = 0; i < irep.plen; i++ ) {
162 int siz = 0;
163 switch( *p++ ) {
164 case IREP_TT_STR:
165 case IREP_TT_SSTR: siz = bin_to_uint16(p) + 3; break;
166 case IREP_TT_INT32: siz = 4; break;
167 case IREP_TT_INT64:
168#if !defined(MRBC_INT64)
169 mrbc_raise(vm, MRBC_CLASS(NotImplementedError), "Unsuported int64 (set MRBC_INT64 in vm_config)");
170#endif
171 case IREP_TT_FLOAT: siz = 8; break;
172 default:
173 mrbc_raise(vm, MRBC_CLASS(Exception), "Loader unknown TT found.");
174 return NULL;
175 }
176 p += siz;
177 }
178
179 // num of symbols, offset of tbl_ireps.
180 irep.slen = bin_to_uint16(p); p += 2;
181 int siz = sizeof(mrbc_sym) * irep.slen + sizeof(uint16_t) * irep.plen;
182 siz += (-siz & 0x03); // padding. 32bit align.
183 irep.ofs_ireps = siz >> 2;
184
185 // allocate new irep
186 mrbc_irep *p_irep;
187 siz = sizeof(mrbc_irep) + siz + sizeof(mrbc_irep*) * irep.rlen;
188 if( vm->vm_id == 0 && !flag_top ) {
189 p_irep = mrbc_raw_alloc_no_free( siz );
190 } else {
191 p_irep = mrbc_raw_alloc( siz );
192 }
193 if( !p_irep ) { // ENOMEM
194 mrbc_raise(vm, MRBC_CLASS(NoMemoryError),0);
195 return NULL;
196 }
197 *p_irep = irep;
198
199 // make a sym_id table.
200 mrbc_sym *tbl_syms = mrbc_irep_tbl_syms(p_irep);
201 for( i = 0; i < irep.slen; i++ ) {
202 int siz = bin_to_uint16(p); p += 2;
203 mrbc_sym sym = mrbc_str_to_symid( (const char *)p );
204 if( sym < 0 ) {
205 mrbc_raise(vm, MRBC_CLASS(Exception), "Overflow MAX_SYMBOLS_COUNT");
206 return NULL;
207 }
208 *tbl_syms++ = sym;
209 p += (siz+1);
210 }
211
212 // make a pool data's offset table.
213 uint16_t *ofs_pools = mrbc_irep_tbl_pools(p_irep);
214 p = p_irep->pool + 2;
215 for( i = 0; i < irep.plen; i++ ) {
216 int siz;
217 *ofs_pools++ = (uint16_t)(p - irep.pool);
218 switch( *p++ ) {
219 case IREP_TT_STR:
220 case IREP_TT_SSTR: siz = bin_to_uint16(p) + 3; break;
221 case IREP_TT_INT32: siz = 4; break;
222 case IREP_TT_INT64:
223 case IREP_TT_FLOAT: siz = 8; break;
224 }
225 p += siz;
226 }
227
228 // return length
229 *len = bin_to_uint32(bin);
230 return p_irep;
231}

242static mrbc_irep *load_irep(struct VM *vm, const uint8_t *bin, int *len)
243{
244 int len1;
245 mrbc_irep *irep = load_irep_1(vm, bin, &len1, len == 0);
246 if( !irep ) return NULL;
247 int total_len = len1;
248
249 mrbc_irep **tbl_ireps = mrbc_irep_tbl_ireps(irep);
250 int i;
251 for( i = 0; i < irep->rlen; i++ ) {
252 tbl_ireps[i] = load_irep(vm, bin + total_len, &len1);
253 if( ! tbl_ireps[i] ) return NULL;
254 total_len += len1;
255 }
256
257 if( len ) *len = total_len;
258 return irep;
259}

271int mrbc_load_mrb(struct VM *vm, const void *bytecode)
272{
273 const uint8_t *bin = bytecode;
274
275 vm->exception = mrbc_nil_value();
276 if( load_header(vm, bin) != 0 ) return -1;
277
278 bin += SIZE_RITE_BINARY_HEADER;
279
280 while( 1 ) {
281 if( memcmp(bin, IREP, sizeof(IREP)) == 0 ) {
282 if( mrbc_load_irep( vm, bin ) != 0 ) break;
283
284 } else if( memcmp(bin, END, sizeof(END)) == 0 ) {
285 break;
286 }
287 // ignore other section.
288
289 bin += bin_to_uint32(bin+4); // add section size, to next section.
290 }
291
292 return mrbc_israised(vm);
293}

340mrbc_value mrbc_irep_pool_value(struct VM *vm, int n)
341{
342 assert( vm->cur_irep->plen > n );
343 const uint8_t *p = mrbc_irep_pool_ptr(vm->cur_irep, n);
344 mrbc_value obj;
345
346 int tt = *p++;
347 switch( tt ) {
348#if MRBC_USE_STRING
349 case IREP_TT_STR:
350 case IREP_TT_SSTR: {
351 int len = bin_to_uint16(p);
352 obj = mrbc_string_new( vm, p+2, len );
353 break;
354 }
355#endif
356
357 case IREP_TT_INT32:
358 mrbc_set_integer(&obj, bin_to_uint32(p));
359 break;
360
361#if MRBC_USE_FLOAT
362 case IREP_TT_FLOAT:
363 mrbc_set_float(&obj, bin_to_double64(p));
364 break;
365#endif
366
367#ifdef MRBC_INT64
368 case IREP_TT_INT64:
369 mrbc_set_integer(&obj, obj.i = bin_to_int64(p));
370 break;
371#endif
372
373 default:
374 mrbc_raisef(vm, MRBC_CLASS(Exception), "Not support such type (IREP_TT=%d)", tt);
375 mrbc_set_nil(&obj);
376 }
377
378 return obj;
379}
