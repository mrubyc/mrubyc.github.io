80mrbc_class * mrbc_define_class(struct VM *vm, const char *name, mrbc_class *super)
81{
82 mrbc_sym sym_id = mrbc_str_to_symid(name);
83 if( sym_id < 0 ) {
84 mrbc_raise(vm, MRBC_CLASS(Exception), "Overflow MAX_SYMBOLS_COUNT");
85 return 0;
86 }
87
88 // already defined?
89 mrbc_value *val = mrbc_get_const(sym_id);
90 if( val ) {
91 assert( mrbc_type(*val) == MRBC_TT_CLASS );
92 return val->cls;
93 }
94
95 // create a new class.
96 mrbc_class *cls = mrbc_raw_alloc_no_free( sizeof(mrbc_class) );
97 if( !cls ) return cls; // ENOMEM
98
99 cls->sym_id = sym_id;
100 cls->num_builtin_method = 0;
101 cls->super = super ? super : mrbc_class_object;
102 cls->method_link = 0;
103
104 if( vm && mrbc_type(vm->cur_regs[0]) == MRBC_TT_CLASS ) {
105 // For nested class
106 // so, not in TOPLEVEL, register to class constant
107 assert(vm->target_class);
108 mrbc_set_class_const(vm->target_class, sym_id,
109 &(mrbc_value){.tt = MRBC_TT_CLASS, .cls = cls});
110 } else {
111 // register to global constant.
112 mrbc_set_const( sym_id, &(mrbc_value){.tt = MRBC_TT_CLASS, .cls = cls} );
113 }
114
115 return cls;
116}

127void mrbc_define_method(struct VM *vm, mrbc_class *cls, const char *name, mrbc_func_t cfunc)
128{
129 if( cls == NULL ) cls = mrbc_class_object; // set default to Object.
130
131 mrbc_method *method = mrbc_raw_alloc_no_free( sizeof(mrbc_method) );
132 if( !method ) return; // ENOMEM
133
134 method->type = 'm';
135 method->c_func = 1;
136 method->sym_id = mrbc_str_to_symid( name );
137 if( method->sym_id < 0 ) {
138 mrbc_raise(vm, MRBC_CLASS(Exception), "Overflow MAX_SYMBOLS_COUNT");
139 }
140 method->func = cfunc;
141 method->next = cls->method_link;
142 cls->method_link = method;
143}

311mrbc_method * mrbc_find_method( mrbc_method *r_method, mrbc_class *cls, mrbc_sym sym_id )
312{
313 do {
314 mrbc_method *method;
315 for( method = cls->method_link; method != 0; method = method->next ) {
316 if( method->sym_id == sym_id ) {
317 *r_method = *method;
318 r_method->cls = cls;
319 return r_method;
320 }
321 }
322
323 struct RBuiltinClass *c = (struct RBuiltinClass *)cls;
324 int right = c->num_builtin_method;
325 if( right == 0 ) goto NEXT;
326 int left = 0;
327
328 while( left < right ) {
329 int mid = (left + right) / 2;
330 if( c->method_symbols[mid] < sym_id ) {
331 left = mid + 1;
332 } else {
333 right = mid;
334 }
335 }
336
337 if( c->method_symbols[right] == sym_id ) {
338 *r_method = (mrbc_method){
339 .type = 'm',
340 .c_func = 2,
341 .sym_id = sym_id,
342 .func = c->method_functions[right],
343 .cls = cls };
344 return r_method;
345 }
346
347 NEXT:
348 cls = cls->super;
349 } while( cls != 0 );
350
351 return 0;
352}

394mrbc_value mrbc_send( struct VM *vm, mrbc_value *v, int reg_ofs,
395 mrbc_value *recv, const char *method_name, int argc, ... )
396{
397 mrbc_method method;
398
399 if( mrbc_find_method( &method, find_class_by_object(recv),
400 mrbc_str_to_symid(method_name) ) == 0 ) {
401 mrbc_printf("No method. vtype=%d method='%s'\n", mrbc_type(*recv), method_name );
402 goto ERROR;
403 }
404 if( !method.c_func ) {
405 mrbc_printf("Method %s needs to be C function.\n", method_name );
406 goto ERROR;
407 }
408
409 // create call stack.
410 mrbc_value *regs = v + reg_ofs + 2;
411 mrbc_decref( &regs[0] );
412 regs[0] = *recv;
413 mrbc_incref(recv);
414
415 va_list ap;
416 va_start(ap, argc);
417 int i;
418 for( i = 1; i <= argc; i++ ) {
419 mrbc_decref( &regs[i] );
420 regs[i] = *va_arg(ap, mrbc_value *);
421 }
422 mrbc_decref( &regs[i] );
423 regs[i] = mrbc_nil_value();
424 va_end(ap);
425
426 // call method.
427 method.func(vm, regs, argc);
428 mrbc_value ret = regs[0];
429
430 for(; i >= 0; i-- ) {
431 regs[i].tt = MRBC_TT_EMPTY;
432 }
433
434 return ret;
435
436 ERROR:
437 return mrbc_nil_value();
438}

456int mrbc_run_mrblib(const void *bytecode)
457{
458 // instead of mrbc_vm_open()
459 mrbc_vm *vm = mrbc_alloc( 0, sizeof(mrbc_vm) );
460 if( !vm ) return -1; // ENOMEM
461 memset(vm, 0, sizeof(mrbc_vm));
462
463 if( mrbc_load_mrb(vm, bytecode) ) {
464 mrbc_print_exception(&vm->exception);
465 return 2;
466 }
467 mrbc_vm_begin(vm);
468 int ret = mrbc_vm_run(vm);
469 mrbc_vm_end(vm);
470
471 // instead of mrbc_vm_close()
472 mrbc_raw_free( vm->top_irep ); // free only top-level mrbc_irep.
473 // (no need to free child ireps.)
474 mrbc_raw_free( vm );
475
476 return ret;
477}
