327 mrbc_value mrbc_send( struct VM *vm, mrbc_value *v, int reg_ofs,
328  mrbc_value *recv, const char *method, int argc, ... )
329 {
330  mrbc_sym sym_id = str_to_symid(method);
331  mrbc_proc *m = find_method(vm, recv, sym_id);
332 
333  if( m == 0 ) {
334  console_printf("No method. vtype=%d method='%s'\n", recv->tt, method );
335  goto ERROR;
336  }
337  if( !m->c_func ) {
338  console_printf("Method %s is not C function\n", method );
339  goto ERROR;
340  }
341 
342  // create call stack.
343  mrbc_value *regs = v + reg_ofs + 2;
344  mrbc_decref( &regs[0] );
345  regs[0] = *recv;
346  mrbc_incref(recv);
347

349  va_start(ap, argc);
350  int i;
351  for( i = 1; i <= argc; i++ ) {
352  mrbc_decref( &regs[i] );
353  regs[i] = *va_arg(ap, mrbc_value *);
354  }
355  mrbc_decref( &regs[i] );
356  regs[i] = mrbc_nil_value();
357  va_end(ap);
358 
359  // call method.
360  m->func(vm, regs, argc);
361  mrbc_value ret = regs[0];
362 
363  for(; i >= 0; i-- ) {
364  regs[i].tt = MRBC_TT_EMPTY;
365  }
366 
367  return ret;
368 
369  ERROR:
370  return mrbc_nil_value();
371 }
