182static inline mrbc_class *find_class_by_object(const mrbc_value *obj)
183{
184 assert( mrbc_type(*obj) >= 0 );
185 assert( mrbc_type(*obj) <= MRBC_TT_MAXVAL );
186
187 mrbc_class *cls = mrbc_class_tbl[ mrbc_type(*obj) ];
188 if( !cls ) {
189 switch( mrbc_type(*obj) ) {
190 case MRBC_TT_CLASS: cls = obj->cls; break;
191 case MRBC_TT_OBJECT: cls = obj->instance->cls; break;
192 case MRBC_TT_EXCEPTION: cls = obj->exception->cls; break;
193 default:
194 assert(!"Invalid value type.");
195 }
196 }
197
198 return cls;
199}
