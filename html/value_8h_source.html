289static inline void mrbc_incref(mrbc_value *v)
290{
291 if( v->tt <= MRBC_TT_INC_DEC_THRESHOLD ) return;
292
293 assert( v->obj->ref_count != 0 );
294 assert( v->obj->ref_count != 0xff ); // check max value.
295 v->obj->ref_count++;
296}

304static inline void mrbc_decref(mrbc_value *v)
305{
306 if( v->tt <= MRBC_TT_INC_DEC_THRESHOLD ) return;
307
308 assert( v->obj->ref_count != 0 );
309 assert( v->obj->ref_count != 0xffff ); // check broken data.
310
311 if( --v->obj->ref_count != 0 ) return;
312
313 (*mrbc_delfunc[v->tt])(v);
314}
