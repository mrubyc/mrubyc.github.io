237 {

239 
240  assert( v->obj->ref_count > 0 );

251 static inline void mrbc_decref(mrbc_value *v)
252 {
253  if( v->tt < MRBC_TT_INC_DEC_THRESHOLD ) return;
254 
255  assert( v->obj->ref_count != 0 );
256  assert( v->obj->ref_count != 0xffff ); // check broken data.
257 
258  if( --v->obj->ref_count != 0 ) return;
259 
260  (*mrbc_delfunc[ v->tt - MRBC_TT_INC_DEC_THRESHOLD ])(v);
261 }
