328static inline void mrbc_incref(mrbc_value *v)
329{
330 if( v->tt <= MRBC_TT_INC_DEC_THRESHOLD ) return;
331
332 assert( v->obj->ref_count != 0 );
333 assert( v->obj->ref_count != 0xff ); // check max value.
334 v->obj->ref_count++;
335}
336
337
338//================================================================
343static inline void mrbc_decref(mrbc_value *v)
344{
345 if( v->tt <= MRBC_TT_INC_DEC_THRESHOLD ) return;
346
347 assert( v->obj->ref_count != 0 );
348 assert( v->obj->ref_count != 0xffff ); // check broken data.
349
350 if( --v->obj->ref_count != 0 ) return;
351
352 (*mrbc_delfunc[v->tt])(v);
353}
