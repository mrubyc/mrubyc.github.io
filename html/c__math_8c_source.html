40static double to_double( struct VM *vm, const mrbc_value *v )
41{
42 switch( mrbc_type(*v) ) {
43 case MRBC_TT_INTEGER: return (double)v->i;
44 case MRBC_TT_FLOAT: return (double)v->d;
45 default: break;
46 }
47
48 mrbc_raise(vm, MRBC_CLASS(TypeError), 0);
49 return 0.0;
50}

167static void c_math_ldexp(struct VM *vm, mrbc_value v[], int argc)
168{
169 int exp;
170 switch( mrbc_type(v[2]) ) {
171 case MRBC_TT_INTEGER: exp = v[2].i; break;
172 case MRBC_TT_FLOAT: exp = (int)v[2].d; break;
173 default:
174 mrbc_raise(vm, MRBC_CLASS(TypeError), 0);
175 return;
176 }
177
178 v[0] = mrbc_float_value(vm, ldexp( to_double(vm, &v[1]), exp ));
179}
