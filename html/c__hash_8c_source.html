137 int mrbc_hash_set(mrbc_value *hash, mrbc_value *key, mrbc_value *val)
138 {

140  int ret = 0;

142  // set a new value
143  if( (ret = mrbc_array_push(hash, key)) != 0 ) goto RETURN;
144  ret = mrbc_array_push(hash, val);
145 
146  } else {

220 int mrbc_hash_compare(const mrbc_value *v1, const mrbc_value *v2)
221 {
222  if( v1->hash->n_stored != v2->hash->n_stored ) return 1;
223 
224  mrbc_value *d1 = v1->hash->data;
225  int i;
226  for( i = 0; i < mrbc_hash_size(v1); i++, d1++ ) {
227  mrbc_value *d2 = mrbc_hash_search(v2, d1); // check key
228  if( d2 == NULL ) return 1;
229  if( mrbc_compare( ++d1, ++d2 ) ) return 1; // check data
230  }
231 
232  return 0;
233 }
