68mrbc_value mrbc_hash_new(struct VM *vm, int size)
69{
70 mrbc_value value = {.tt = MRBC_TT_HASH};
71
72 /*
73 Allocate handle and data buffer.
74 */
75 mrbc_hash *h = mrbc_alloc(vm, sizeof(mrbc_hash));
76 if( !h ) return value; // ENOMEM
77
78 mrbc_value *data = mrbc_alloc(vm, sizeof(mrbc_value) * size * 2);
79 if( !data ) { // ENOMEM
80 mrbc_raw_free( h );
81 return value;
82 }
83
84 MRBC_INIT_OBJECT_HEADER( h, "HA" );
85 h->data_size = size * 2;
86 h->n_stored = 0;
87 h->data = data;
88
89 value.hash = h;
90 return value;
91}

114mrbc_value * mrbc_hash_search(const mrbc_value *hash, const mrbc_value *key)
115{
116#ifndef MRBC_HASH_SEARCH_LINER
117#define MRBC_HASH_SEARCH_LINER
118#endif
119
120#ifdef MRBC_HASH_SEARCH_LINER
121 mrbc_value *p1 = hash->hash->data;
122 const mrbc_value *p2 = p1 + hash->hash->n_stored;
123
124 while( p1 < p2 ) {
125 if( mrbc_compare(p1, key) == 0 ) return p1;
126 p1 += 2;
127 }
128 return NULL;
129#endif
130
131#ifdef MRBC_HASH_SEARCH_LINER_ITERATOR
132 mrbc_hash_iterator ite = mrbc_hash_iterator_new(hash);
133 while( mrbc_hash_i_has_next(&ite) ) {
134 mrbc_value *v = mrbc_hash_i_next(&ite);
135 if( mrbc_compare( v, key ) == 0 ) return v;
136 }
137 return NULL;
138#endif
139}

150int mrbc_hash_set(mrbc_value *hash, mrbc_value *key, mrbc_value *val)
151{
152 mrbc_value *v = mrbc_hash_search(hash, key);
153 int ret = 0;
154 if( v == NULL ) {
155 // set a new value
156 if( (ret = mrbc_array_push(hash, key)) != 0 ) goto RETURN;
157 ret = mrbc_array_push(hash, val);
158
159 } else {
160 // replace a value
161 mrbc_decref(v);
162 *v = *key;
163 mrbc_decref(++v);
164 *v = *val;
165 }
166
167 RETURN:
168 return ret;
169}

193mrbc_value mrbc_hash_remove(mrbc_value *hash, mrbc_value *key)
194{
195 mrbc_value *v = mrbc_hash_search(hash, key);
196 if( v == NULL ) return mrbc_nil_value();
197
198 mrbc_decref(v); // key
199 mrbc_value val = v[1]; // value
200
201 mrbc_hash *h = hash->hash;
202 h->n_stored -= 2;
203
204 memmove(v, v+2, (char*)(h->data + h->n_stored) - (char*)v);
205
206 // TODO: re-index hash table if need.
207
208 return val;
209}

233int mrbc_hash_compare(const mrbc_value *v1, const mrbc_value *v2)
234{
235 if( v1->hash->n_stored != v2->hash->n_stored ) return 1;
236
237 mrbc_value *d1 = v1->hash->data;
238 int i;
239 for( i = 0; i < mrbc_hash_size(v1); i++, d1++ ) {
240 mrbc_value *d2 = mrbc_hash_search(v2, d1); // check key
241 if( d2 == NULL ) return 1;
242 if( mrbc_compare( ++d1, ++d2 ) ) return 1; // check data
243 }
244
245 return 0;
246}

255mrbc_value mrbc_hash_dup( struct VM *vm, mrbc_value *src )
256{
257 mrbc_value ret = mrbc_hash_new(vm, mrbc_hash_size(src));
258 if( ret.hash == NULL ) return ret; // ENOMEM
259
260 mrbc_hash *h = src->hash;
261 memcpy( ret.hash->data, h->data, sizeof(mrbc_value) * h->n_stored );
262 ret.hash->n_stored = h->n_stored;
263
264 mrbc_value *p1 = h->data;
265 const mrbc_value *p2 = p1 + h->n_stored;
266 while( p1 < p2 ) {
267 mrbc_incref(p1++);
268 }
269
270 // TODO: dup other members.
271
272 return ret;
273}
