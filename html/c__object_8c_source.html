42static void c_object_new(struct VM *vm, mrbc_value v[], int argc)
43{
44 mrbc_class *cls = v->cls;
45 mrbc_value new_obj = mrbc_instance_new(vm, cls, 0);
46 if( new_obj.instance == NULL ) return; // ENOMEM
47
48 mrbc_decref(&v[0]);
49 v[0] = new_obj;
50
51 // call the initialize method.
52 mrbc_method method;
53 if( mrbc_find_method( &method, cls, MRBC_SYM(initialize) ) == NULL ) return;
54
55 mrbc_decref(&v[argc+1]);
56 mrbc_set_nil(&v[argc+1]);
57 mrbc_callinfo *callinfo = mrbc_push_callinfo(vm, MRBC_SYM(initialize),
58 (v - vm->cur_regs), argc);
59 callinfo->own_class = method.cls;
60
61 vm->cur_irep = method.irep;
62 vm->inst = vm->cur_irep->inst;
63 vm->cur_regs = v;
64}

262static void c_object_raise(struct VM *vm, mrbc_value v[], int argc)
263{
264 assert( !mrbc_israised(vm) );
265
266 // case 1. raise (no argument)
267 if( argc == 0 ) {
268 vm->exception = mrbc_exception_new( vm, MRBC_CLASS(RuntimeError), "", 0 );
269 } else
270
271 // case 2. raise "message"
272 if( argc == 1 && mrbc_type(v[1]) == MRBC_TT_STRING ) {
273 vm->exception = mrbc_exception_new( vm, MRBC_CLASS(RuntimeError),
274 mrbc_string_cstr(&v[1]), mrbc_string_size(&v[1]) );
275 } else
276
277 // case 3. raise ExceptionClass
278 if( argc == 1 && mrbc_type(v[1]) == MRBC_TT_CLASS &&
279 mrbc_obj_is_kind_of( &v[1], MRBC_CLASS(Exception))) {
280 vm->exception = mrbc_exception_new( vm, v[1].cls, 0, 0 );
281 } else
282
283 // case 4. raise ExceptionObject
284 if( argc == 1 && mrbc_type(v[1]) == MRBC_TT_EXCEPTION ) {
285 mrbc_incref( &v[1] );
286 vm->exception = v[1];
287 } else
288
289 // case 5. raise ExceptionClass, "param"
290 if( argc == 2 && mrbc_type(v[1]) == MRBC_TT_CLASS
291 && mrbc_type(v[2]) == MRBC_TT_STRING ) {
292 vm->exception = mrbc_exception_new( vm, v[1].cls,
293 mrbc_string_cstr(&v[2]), mrbc_string_size(&v[2]) );
294 } else
295
296 // case 6. raise ExceptionObject, "param"
297 if( argc == 2 && mrbc_type(v[1]) == MRBC_TT_EXCEPTION
298 && mrbc_type(v[2]) == MRBC_TT_STRING ) {
299 vm->exception = mrbc_exception_new( vm, v[1].exception->cls,
300 mrbc_string_cstr(&v[2]), mrbc_string_size(&v[2]) );
301 } else {
302
303 // fail.
304 vm->exception = mrbc_exception_new( vm, MRBC_CLASS(ArgumentError), 0, 0 );
305 }
306
307 vm->flag_preemption = 2;
308}

489//================================================================
492static void c_object_sprintf(struct VM *vm, mrbc_value v[], int argc)
493{
494 static const int BUF_INC_STEP = 32; // bytes.
495
496 mrbc_value *format = &v[1];
497 if( mrbc_type(*format) != MRBC_TT_STRING ) {
498 mrbc_raise(vm, MRBC_CLASS(TypeError), "sprintf");
499 return;
500 }
501
502 int buflen = BUF_INC_STEP;
503 char *buf = mrbc_alloc(vm, buflen);
504 if( !buf ) { return; } // ENOMEM raise?
505
506 mrbc_printf_t pf;
507 mrbc_printf_init( &pf, buf, buflen, mrbc_string_cstr(format) );
508
509 int i = 2;
510 int ret;
511 while( 1 ) {
512 mrbc_printf_t pf_bak = pf;
513 ret = mrbc_printf_main( &pf );
514 if( ret == 0 ) break; // normal break loop.
515 if( ret < 0 ) goto INCREASE_BUFFER;
516
517 if( i > argc ) {
518 mrbc_raise(vm, MRBC_CLASS(ArgumentError), "too few arguments");
519 break;
520 }
521
522 // maybe ret == 1
523 switch(pf.fmt.type) {
524 case 'c':
525 if( mrbc_type(v[i]) == MRBC_TT_INTEGER ) {
526 ret = mrbc_printf_char( &pf, v[i].i );
527 } else if( mrbc_type(v[i]) == MRBC_TT_STRING ) {
528 ret = mrbc_printf_char( &pf, mrbc_string_cstr(&v[i])[0] );
529 }
530 break;
531
532 case 's':
533 if( mrbc_type(v[i]) == MRBC_TT_STRING ) {
534 ret = mrbc_printf_bstr( &pf, mrbc_string_cstr(&v[i]), mrbc_string_size(&v[i]),' ');
535 } else if( mrbc_type(v[i]) == MRBC_TT_SYMBOL ) {
536 ret = mrbc_printf_str( &pf, mrbc_symbol_cstr( &v[i] ), ' ');
537 }
538 break;
539
540 case 'd':
541 case 'i':
542 case 'u':
543 if( mrbc_type(v[i]) == MRBC_TT_INTEGER ) {
544 ret = mrbc_printf_int( &pf, v[i].i, 10);
545#if MRBC_USE_FLOAT
546 } else if( mrbc_type(v[i]) == MRBC_TT_FLOAT ) {
547 ret = mrbc_printf_int( &pf, (mrbc_int_t)v[i].d, 10);
548#endif
549 } else if( mrbc_type(v[i]) == MRBC_TT_STRING ) {
550 mrbc_int_t ival = atol(mrbc_string_cstr(&v[i]));
551 ret = mrbc_printf_int( &pf, ival, 10 );
552 }
553 break;
554
555 case 'b':
556 case 'B':
557 if( mrbc_type(v[i]) == MRBC_TT_INTEGER ) {
558 ret = mrbc_printf_bit( &pf, v[i].i, 1);
559 }
560 break;
561
562 case 'x':
563 case 'X':
564 if( mrbc_type(v[i]) == MRBC_TT_INTEGER ) {
565 ret = mrbc_printf_bit( &pf, v[i].i, 4);
566 }
567 break;
568
569 case 'o':
570 if( mrbc_type(v[i]) == MRBC_TT_INTEGER ) {
571 ret = mrbc_printf_bit( &pf, v[i].i, 3);
572 }
573 break;
574
575#if MRBC_USE_FLOAT
576 case 'f':
577 case 'e':
578 case 'E':
579 case 'g':
580 case 'G':
581 if( mrbc_type(v[i]) == MRBC_TT_FLOAT ) {
582 ret = mrbc_printf_float( &pf, v[i].d );
583 } else if( mrbc_type(v[i]) == MRBC_TT_INTEGER ) {
584 ret = mrbc_printf_float( &pf, v[i].i );
585 }
586 break;
587#endif
588
589 default:
590 break;
591 }
592 if( ret >= 0 ) {
593 i++;
594 continue; // normal next loop.
595 }
596
597 // maybe buffer full. (ret == -1)
598 if( pf.fmt.width > BUF_INC_STEP ) buflen += pf.fmt.width;
599 pf = pf_bak;
600
601 INCREASE_BUFFER:
602 buflen += BUF_INC_STEP;
603 buf = mrbc_realloc(vm, pf.buf, buflen);
604 if( !buf ) { return; } // ENOMEM raise? TODO: leak memory.
605 mrbc_printf_replace_buffer(&pf, buf, buflen);
606 }
607 mrbc_printf_end( &pf );
608
609 buflen = mrbc_printf_len( &pf );
610 mrbc_realloc(vm, pf.buf, buflen+1); // shrink suitable size.
611
612 mrbc_value value = mrbc_string_new_alloc( vm, pf.buf, buflen );
613
614 SET_RETURN(value);
615}
