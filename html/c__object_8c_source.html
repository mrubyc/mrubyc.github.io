48 {
49  mrbc_value new_obj = mrbc_instance_new(vm, v->cls, 0);

75  };

77  mrbc_class *cls = v->cls;

80  v[0] = new_obj;

281 static void c_object_raise(struct VM *vm, mrbc_value v[], int argc)
282 {
283  if( !vm->exc ){
284  // raise exception
285  if( argc == 0 ){
286  // 1. raise
287  vm->exc = mrbc_class_runtimeerror;
288  vm->exc_message = mrbc_nil_value();
289  } else if( argc == 1 ){
290  if( v[1].tt == MRBC_TT_CLASS ){
291  // 3. raise Exception
292  vm->exc = v[1].cls;
293  vm->exc_message = mrbc_nil_value();
294  } else {
295  // 2. raise "param"
296  mrbc_incref( &v[1] );
297  vm->exc = mrbc_class_runtimeerror;
298  vm->exc_message = v[1];
299  }
300  } else if( argc == 2 ){
301  // 4. raise Exception, "param"
302  mrbc_incref( &v[2] );
303  vm->exc = v[1].cls;
304  vm->exc_message = v[2];
305  }
306  } else {
307  // in exception
308  }
309 
310  // do nothing if no rescue, no ensure
311  if( vm->exception_tail == NULL ){
312  return;
313  }
314 
315  // NOT to return to OP_SEND
316  mrbc_pop_callinfo(vm);
317 
318  mrbc_callinfo *callinfo = vm->exception_tail;
319  if( callinfo != NULL ){
320  if( callinfo->method_id == 0x7fff ){
321  // "rescue"
322  // jump to rescue
323  vm->exception_tail = callinfo->prev;
324  vm->current_regs = callinfo->current_regs;
325  vm->pc_irep = callinfo->pc_irep;
326  vm->inst = callinfo->inst;
327  vm->target_class = callinfo->target_class;
328  mrbc_free(vm, callinfo);
329  callinfo = vm->exception_tail;
330  } else {
331  // "ensure"
332  // jump to ensure
333  vm->exception_tail = callinfo->prev;
334  vm->current_regs = callinfo->current_regs;
335  vm->pc_irep = callinfo->pc_irep;
336  vm->inst = callinfo->inst;
337  vm->target_class = callinfo->target_class;
338  mrbc_free(vm, callinfo);
339  //
340  callinfo = vm->exception_tail;
341  if( callinfo != NULL ){
342  vm->exception_tail = callinfo->prev;
343  callinfo->prev = vm->callinfo_tail;
344  vm->callinfo_tail = callinfo;
345  }
346  }
347  }
348  if( callinfo == NULL ){
349  vm->exc_pending = vm->exc;
350  vm->exc = 0;
351  }
352 }

549 
550  case 's':

759 {

761

763  mrbc_callinfo *callinfo = mrbc_push_callinfo(vm,
